005    package com.nuecho.rivr.core.servlet.session;
006    
007    import javax.servlet.http.*;
008    
009    import com.nuecho.rivr.core.channel.*;
010    import com.nuecho.rivr.core.channel.synchronous.*;
011    import com.nuecho.rivr.core.dialogue.*;
012    import com.nuecho.rivr.core.servlet.*;
013    import com.nuecho.rivr.core.util.*;
014    
015    /**
016     * @author Nu Echo Inc.
017     */
018    public final class Session<I extends InputTurn, O extends OutputTurn, F extends FirstTurn, L extends LastTurn, C extends DialogueContext<I, O>>
019            implements DialogueChannelListener<I, O> {
020        private SynchronousDialogueChannel<I, O, F, L, C> mDialogueChannel;
021    
022        private C mDialogueContext;
023    
024        private final SessionContainer<I, O, F, L, C> mContainer;
025        private final String mId;
026        private HttpSession mAssociatedHttpSession;
027    
028        public Session(SessionContainer<I, O, F, L, C> container, String sessionId) {
029            mContainer = container;
030            mId = sessionId;
031        }
032    
033        @Override
034        public void onStart(DialogueChannel<I, O> dialogueChannel) {}
035    
036        @Override
037        public void onStop(DialogueChannel<I, O> dialogueChannel) {
038            stop();
039        }
040    
041        public void stop() {
042            if (mDialogueChannel != null && mDialogueChannel.isDialogueActive()) {
043                mDialogueChannel.stop();
044            }
045    
046            mContainer.removeSession(mId);
047    
048            if (mAssociatedHttpSession != null) {
049                mAssociatedHttpSession.invalidate();
050                mAssociatedHttpSession = null;
051            }
052        }
053    
054        public void keepAlive() {
055            if (mAssociatedHttpSession != null) {
056                mAssociatedHttpSession.getAttributeNames();
057            }
058        }
059    
060        public String getId() {
061            return mId;
062        }
063    
064        public SynchronousDialogueChannel<I, O, F, L, C> getDialogueChannel() {
065            return mDialogueChannel;
066        }
067    
068        public void setDialogueChannel(SynchronousDialogueChannel<I, O, F, L, C> dialogueChannel) {
069            mDialogueChannel = dialogueChannel;
070            mDialogueChannel.addListener(this);
071        }
072    
073        public C getDialogueContext() {
074            return mDialogueContext;
075        }
076    
077        public void setDialogueContext(C dialogueContext) {
078            mDialogueContext = dialogueContext;
079        }
080    
081        public void setAssociatedHttpSession(HttpSession associatedHttpSession) {
082            mAssociatedHttpSession = associatedHttpSession;
083        }
084    
085        @Override
086        public String toString() {
087            ToStringBuilder builder = new ToStringBuilder(this);
088            builder.appendItem("mDialogueChannel", mDialogueChannel);
089            builder.appendItem("mDialogueContext", mDialogueContext);
090            builder.appendItem("mId", mId);
091            return builder.getString();
092        }
093    }
