052        @Override
053        public final void init() throws ServletException {
054    
055            try {
056                initDialogueServlet();
057            } catch (DialogueServletInitializationException exception) {
058                throw new ServletException("Unable to initialize dialogue servlet.s", exception);
059            }
060    
061            if (mLoggerFactory == null) {
062                mLoggerFactory = LoggerFactory.getILoggerFactory();
063            }
064    
065            ensureFieldIsSet(mInputTurnFactory, "InputTurnFactory");
066            ensureFieldIsSet(mDialogueFactory, "DialogueFactory");
067            ensureFieldIsSet(mDialogueContextFactory, "DialogueContextFactory");
068            ensureFieldIsSet(mErrorHandler, "ErrorHandler");
069    
070            Logger logger = mLoggerFactory.getLogger(SESSION_LOGGER_NAME);
071    
072            mSessionContainer = new SessionContainer<I, O, F, L, C>(logger,
073                                                                    mSessionTimeout,
074                                                                    mSessionScanPeriod,
075                                                                    SESSION_CONTAINER_NAME);
076    
077        }

081        }

098        public void renderOutputTurn(O outputTurn,
099                                     HttpServletRequest request,
100                                     final HttpServletResponse response,
101                                     Session<I, O, F, L, C> session) throws IOException, StepRendererException {

103                                                                                                                   request,
104                                                                                                                   response,
105                                                                                                                   session.getDialogueContext());

107        }

176        private void process(HttpServletRequest request, HttpServletResponse response) throws ServletException {
177            Session<I, O, F, L, C> session;
178            try {
179                session = getSession(request);
180                MDC.put(MDC_KEY_DIALOGUE_ID, session.getId());
181            } catch (SessionNotFoundException exception) {
182                throw new ServletException("Cannot find session.", exception);
183            }
184    
185            process(request, response, session);
186        }
187    
188        private void process(HttpServletRequest request, HttpServletResponse response, Session<I, O, F, L, C> session)
189                throws ServletException {
190            try {
191    
192                Step<O, L> step;
193                C dialogueContext = session.getDialogueContext();
194    
195                try {
196                    if (dialogueContext == null) {
197                        step = startDialogue(request, response, session);
198                    } else {
199                        step = continueDialogue(request, response, session);
200                    }
201                } catch (Timeout exception) {
202                    renderError(exception, request, response, session);
203                    return;
204                } catch (InterruptedException exception) {
205                    Thread.currentThread().interrupt();
206                    renderError(exception, request, response, session);
207                    return;
208                }
209    
210                if (step instanceof OutputTurnStep) {
211                    OutputTurnStep<O, L> outputTurnStep = (OutputTurnStep<O, L>) step;
212                    renderOutputTurn(outputTurnStep.getOutputTurn(), request, response, session);
213                } else if (step instanceof LastTurnStep) {
214                    LastTurnStep<O, L> lastTurnStep = (LastTurnStep<O, L>) step;
215                    renderLastTurn(lastTurnStep.getLastTurn(), request, response, session);
216                } else if (step instanceof ErrorStep) {
217                    ErrorStep<O, L> errorStep = (ErrorStep<O, L>) step;
218                    Throwable throwable = errorStep.getThrowable();
219                    renderError(throwable, request, response, session);
220                }
221            } catch (Exception exception) {
222                throw new ServletException("Error while rendering step.", exception);
223            }
224        }
225    
226        private Step<O, L> continueDialogue(HttpServletRequest request,
227                                            HttpServletResponse response,
228                                            Session<I, O, F, L, C> session) throws ServletException, Timeout,
229                InterruptedException {
230            Assert.notNull(session, "session");
231            I inputTurn = createInputTurn(request, response);
232            SynchronousDialogueChannel<I, O, F, L, C> dialogueChannel = session.getDialogueChannel();
233            Assert.notNull(dialogueChannel, "dialogueChannel");
234            return dialogueChannel.doTurn(inputTurn, mDialogueTimeout);
235        }
236    
237        private Step<O, L> startDialogue(HttpServletRequest request,
238                                         HttpServletResponse response,
239                                         Session<I, O, F, L, C> session) throws ServletException, Timeout,
240                InterruptedException {
241            SynchronousDialogueChannel<I, O, F, L, C> dialogueChannel;
242            dialogueChannel = new SynchronousDialogueChannel<I, O, F, L, C>();
243            session.setDialogueChannel(dialogueChannel);
244    
245            Logger logger = mLoggerFactory.getLogger(DIALOGUE_LOGGER_NAME);
246            dialogueChannel.setLogger(logger);
247    
248            C dialogueContext = createContext(request, session, dialogueChannel, logger);
249    
250            DialogueInitializationInfo<I, O, C> initializationInfo;
251            initializationInfo = createInitializationInfo(request, response, dialogueContext);
252            Dialogue<I, O, F, L, C> dialogue;
253            try {
254                dialogue = mDialogueFactory.create(initializationInfo);
255            } catch (DialogueFactoryException exception) {
256                throw new ServletException("Unable to create dialogue.", exception);
257            }
258            F firstTurn = createFirstTurn(request, response);
259            return dialogueChannel.start(dialogue, firstTurn, mDialogueTimeout, dialogueContext);
260        }
261    
262        private C createContext(HttpServletRequest request,
263                                Session<I, O, F, L, C> session,
264                                SynchronousDialogueChannel<I, O, F, L, C> dialogueChannel,
265                                Logger logger) {
266            C dialogueContext = mDialogueContextFactory.createDialogueContext(request,
267                                                                              session.getId(),
268                                                                              dialogueChannel,
269                                                                              logger);
270            session.setDialogueContext(dialogueContext);
271            return dialogueContext;
272        }

275                                                                                HttpServletResponse response,
276                                                                                C dialogueContext) {
277            return new WebDialogueInitializationInfo<I, O, C>(dialogueContext, request, response, getServletContext(), this);
278        }

280        protected Session<I, O, F, L, C> getSession(HttpServletRequest request) throws SessionNotFoundException {
281            String pathInfo = request.getPathInfo();
282    
283            if (pathInfo != null && !pathInfo.equals("/")) {
284                if (pathInfo.startsWith("/")) {
285                    pathInfo = pathInfo.substring(1);
286                }
287    
288                int firstSlash = pathInfo.indexOf('/');
289                if (firstSlash != -1) {
290                    pathInfo = pathInfo.substring(0, firstSlash);
291                }
292    
293                return getExistingSession(pathInfo);
294            } else {
295                String sessionId = UUID.randomUUID().toString();
296    
297                Session<I, O, F, L, C> session = new Session<I, O, F, L, C>(mSessionContainer, sessionId);
298                mSessionContainer.addSession(session);
299                return session;
300            }
301        }
302    
303        protected Session<I, O, F, L, C> getExistingSession(String sessionId) throws SessionNotFoundException {
304            Session<I, O, F, L, C> session = mSessionContainer.getSession(sessionId);
305    
306            if (session == null) throw new SessionNotFoundException("Unable to find session [" + sessionId + "]");
307    
308            return session;
309        }

313                return mInputTurnFactory.createInputTurn(request, response);

317        }

321                return mInputTurnFactory.createFirstTurn(request, response);
322            } catch (InputTurnFactoryException exception) {
323                throw new ServletException(exception);

325        }

327        private static void commitToResponse(final HttpServletResponse response, ServletResponseContent responseContent)
328                throws IOException {
329            ServletOutputStream outputStream = response.getOutputStream();
330            response.setContentType(responseContent.getContentType());
331            responseContent.writeTo(outputStream);
332        }
