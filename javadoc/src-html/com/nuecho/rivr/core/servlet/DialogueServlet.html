022/**
023 * Abstract servlet interacting with a web client acting as the controller of a
024 * {@link SynchronousDialogueChannel}.
025 * <p>
026 * This abstract servlet must be extended in order to provide a specific
027 * implementation. For each session,
028 * <ol>
029 * <li>the servlet creates the {@link Session} and place it in the
030 * {@link SessionContainer}</li>
031 * <li>it creates a {@link Dialogue} with a {@link DialogueFactory}</li>
032 * <li>it creates a {@link DialogueContext} with a
033 * {@link DialogueContextFactory}</li>
034 * <li>it creates a {@link SynchronousDialogueChannel} and starts the dialogue
035 * upon initial HTTP request</li>
036 * <li>it renders the various {@link Step steps} from the dialogue channel into
037 * appropriate HTTP responses</li>
038 * <li>it translates HTTP requests into {@link InputTurn InputTurns} using the
039 * {@link InputTurnFactory}
040 * <li>once the dialogue is done, the servlet perform necessary clean-up.</li>
041 * </ol>
042 * <h3>init args</h3> The following servlet initial arguments are supported:
043 * <p/>
044 * <dt>com.nuecho.rivr.core.dialogueTimeout
045 * <dd>Maximum time for dialogue to produce an {@link OutputTurn}. Value
046 * specified must be followed by unit (ms, s, m, h, d, y), e.g. <code>10s</code>
047 * for 10 seconds. Default value: <code>10 s</code></dd></dt>
048 * <p/>
049 * <dt>com.nuecho.rivr.core.controllerTimeout
050 * <dd>Maximum time for controller to produce an {@link InputTurn}. Value
051 * specified must be followed by unit (ms, s, m, h, d, y), e.g. <code>10s</code>
052 * for 10 seconds. Default value: <code>5 m</code></dd></dt>
053 * <p/>
054 * <dt>com.nuecho.rivr.core.sessionTimeout</dt>
055 * <dd>Maximum inactivity time for a session. Value specified must be followed
056 * by unit (ms, s, m, h, d, y), e.g. <code>10s</code> for 10 seconds. Default
057 * value: <code>30 m</code></dd>
058 * <p/>
059 * <dt>com.nuecho.rivr.core.sessionScanPeriod</dt>
060 * <dd>Time between each scan for dead sessions in the session container. Value
061 * specified must be followed by unit (ms, s, m, h, d, y), e.g. <code>10s</code>
062 * for 10 seconds. Default value: <code>2 m</code></dd>
063 * <p/>
064 * <dt>com.nuecho.rivr.core.webappServerSessionTrackingEnabled</dt>
065 * <dd>Whether a {@link javax.servlet.http.HttpSession} should be created for
066 * each dialogue or not. This is useful for load-balancers using JSESSIONID
067 * cookie to enforce server affinity (or stickyness). Value should be
068 * <code>true</code> or <code>false</code>. Default value: <code>true</code>.
069 * 
070 * @param <F> type of {@link FirstTurn}
071 * @param <L> type of {@link LastTurn}
072 * @param <O> type of {@link OutputTurn}
073 * @param <I> type of {@link InputTurn}
074 * @param <C> type of {@link DialogueContext}
075 * @author Nu Echo Inc.
076 */
077public abstract class DialogueServlet<I extends InputTurn, O extends OutputTurn, F extends FirstTurn, L extends LastTurn, C extends DialogueContext<I, O>>
078        extends HttpServlet {

138    /**
139     * Initializes the servlet. The first thing done in this method is to call
140     * {@link #initDialogueServlet()}. This method is called by the servlet
141     * container.
142     */
143    @Override
144    public final void init() throws ServletException {
145
146        Throwable initError = null;
147
148        try {
149            initDialogueServlet();
150        } catch (DialogueServletInitializationException exception) {
151            initError = exception;
152        }
153
154        if (mLoggerFactory == null) {
155            mLoggerFactory = LoggerFactory.getILoggerFactory();
156        }
157
158        mLogger = mLoggerFactory.getLogger(SERVLET_LOGGER_NAME);
159        mResponseLogger = mLoggerFactory.getLogger(RESPONSES_LOGGER_NAME);
160
161        if (initError != null) {
162            mLogger.error("Unable to initialize dialogue servlet.", initError);
163            destroy();
164            throw new ServletException("Unable to initialize dialogue servlet.", initError);
165        }
166
167        ensureFieldIsSet(mInputTurnFactory, "InputTurnFactory");
168        ensureFieldIsSet(mDialogueFactory, "DialogueFactory");
169        ensureFieldIsSet(mDialogueContextFactory, "DialogueContextFactory");
170        ensureFieldIsSet(mErrorHandler, "ErrorHandler");
171
172        Logger sessionContainerLogger = mLoggerFactory.getLogger(SESSION_LOGGER_NAME);
173
174        mSessionContainer = new SessionContainer<I, O, F, L, C>(sessionContainerLogger,
175                                                                mSessionTimeout,
176                                                                mSessionScanPeriod,
177                                                                SESSION_CONTAINER_NAME);
178
179        Duration sessionScanPeriod = getDuration(INITIAL_ARGUMENT_SESSION_SCAN_PERIOD);
180        if (sessionScanPeriod != null) {
181            setSessionScanPeriod(sessionScanPeriod);
182        }
183
184        Duration sessionTimeout = getDuration(INITIAL_ARGUMENT_SESSION_TIMEOUT);
185        if (sessionTimeout != null) {
186            setSessionTimeout(sessionTimeout);
187        }
188
189        Duration dialogueTimeout = getDuration(INITIAL_ARGUMENT_DIALOGUE_TIMEOUT);
190        if (dialogueTimeout != null) {
191            setDialogueTimeout(dialogueTimeout);
192        }
193
194        Duration controllerTimeout = getDuration(INITIAL_ARGUMENT_CONTROLLER_TIMEOUT);
195        if (controllerTimeout != null) {
196            setControllerTimeout(controllerTimeout);
197        }
198
199        Boolean enableWebappServerSessionTracking = getBoolean(INITIAL_ARGUMENT_ENABLE_WEBAPP_SERVER_SESSION_TRACKING);
200        if (enableWebappServerSessionTracking != null) {
201            setWebappServerSessionTrackingEnabled(mWebappServerSessionTrackingEnabled);
202        }
203
204        mLogger.info("Dialogue servlet initialized.");
205
206    }

208    /**
209     * Destroys the servlet. This methods calls
210     * {@link #destroyDialogueServlet()}. This method is called by the servlet
211     * container.
212     */
213    @Override
214    public final synchronized void destroy() {
215        if (mDestroyed) return;
216        if (mSessionContainer != null) {
217            mSessionContainer.stop();
218        }
219        destroyDialogueServlet();
220
221        mLogger.info("Dialogue servlet destroyed.");
222        mDestroyed = true;
223    }

225    private Duration getDuration(String key) throws ServletException {
226        ServletConfig servletConfig = getServletConfig();
227        String duration = servletConfig.getInitParameter(key);
228        if (duration == null) return null;
229        try {
230            return Duration.parse(duration);
231        } catch (IllegalArgumentException exception) {
232            throw new ServletException("Unable to parse duration for init-arg '" + key + "'", exception);
233        }
234    }
235
236    private Boolean getBoolean(String key) throws ServletException {
237        ServletConfig servletConfig = getServletConfig();
238        String booleanString = servletConfig.getInitParameter(key);
239        if (booleanString == null) return null;
240        if (booleanString.equalsIgnoreCase(TRUE)) return Boolean.TRUE;
241        if (booleanString.equalsIgnoreCase(FALSE)) return Boolean.FALSE;
242        throw new ServletException("Unable to parse boolean for init-arg '"
243                                   + key
244                                   + "'.  Should be '"
245                                   + TRUE
246                                   + "' of '"
247                                   + FALSE
248                                   + "' but not '"
249                                   + booleanString
250                                   + "'.");
251    }

373    private void process(HttpServletRequest request, HttpServletResponse response) throws ServletException {
374        Session<I, O, F, L, C> session;
375        try {
376            session = getSession(request);
377            MDC.put(MDC_KEY_DIALOGUE_ID, session.getId());
378            process(request, response, session);
379        } catch (SessionNotFoundException exception) {
380            throw new ServletException("Cannot find session.", exception);
381        } finally {
382            MDC.remove(MDC_KEY_DIALOGUE_ID);
383        }
384
385    }
386
387    private void process(HttpServletRequest request, HttpServletResponse response, Session<I, O, F, L, C> session)
388            throws ServletException {
389        try {
390
391            Step<O, L> step;
392            C dialogueContext = session.getDialogueContext();
393
394            try {
395                if (dialogueContext == null) {
396                    step = startDialogue(request, response, session);
397                } else {
398                    step = continueDialogue(request, response, session);
399                }
400            } catch (Timeout exception) {
401                renderError(exception, request, response, session);
402                return;
403            } catch (InterruptedException exception) {
404                Thread.currentThread().interrupt();
405                renderError(exception, request, response, session);
406                return;
407            }
408
409            if (step instanceof OutputTurnStep) {
410                OutputTurnStep<O, L> outputTurnStep = (OutputTurnStep<O, L>) step;
411                renderOutputTurn(outputTurnStep.getOutputTurn(), request, response, session);
412            } else if (step instanceof LastTurnStep) {
413                LastTurnStep<O, L> lastTurnStep = (LastTurnStep<O, L>) step;
414                renderLastTurn(lastTurnStep.getLastTurn(), request, response, session);
415            } else if (step instanceof ErrorStep) {
416                ErrorStep<O, L> errorStep = (ErrorStep<O, L>) step;
417                Throwable throwable = errorStep.getThrowable();
418                renderError(throwable, request, response, session);
419            }
420        } catch (Exception exception) {
421            throw new ServletException("Error while rendering step.", exception);
422        }
423    }

425    private Step<O, L> continueDialogue(HttpServletRequest request,
426                                        HttpServletResponse response,
427                                        Session<I, O, F, L, C> session) throws ServletException, Timeout,
428            InterruptedException {
429        Assert.notNull(session, "session");
430        I inputTurn = createInputTurn(request, response);
431        SynchronousDialogueChannel<I, O, F, L, C> dialogueChannel = session.getDialogueChannel();
432        Assert.notNull(dialogueChannel, "dialogueChannel");
433        return dialogueChannel.doTurn(inputTurn, mDialogueTimeout);
434    }

436    private Step<O, L> startDialogue(HttpServletRequest request,
437                                     HttpServletResponse response,
438                                     Session<I, O, F, L, C> session) throws ServletException, Timeout,
439            InterruptedException {
440        SynchronousDialogueChannel<I, O, F, L, C> dialogueChannel;
441        dialogueChannel = new SynchronousDialogueChannel<I, O, F, L, C>();
442        session.setDialogueChannel(dialogueChannel);
443
444        Logger logger = mLoggerFactory.getLogger(DIALOGUE_LOGGER_NAME);
445        dialogueChannel.setLogger(logger);
446
447        dialogueChannel.setDefaultReceiveFromControllerTimeout(mControllerTimeout);
448        dialogueChannel.setDefaultReceiveFromDialogueTimeout(mDialogueTimeout);
449
450        C dialogueContext = createContext(request, session, dialogueChannel, logger);
451
452        DialogueInitializationInfo<I, O, C> initializationInfo;
453        initializationInfo = createInitializationInfo(request, response, dialogueContext);
454        Dialogue<I, O, F, L, C> dialogue;
455        try {
456            dialogue = mDialogueFactory.create(initializationInfo);
457        } catch (DialogueFactoryException exception) {
458            throw new ServletException("Unable to create dialogue.", exception);
459        }
460        F firstTurn = createFirstTurn(request, response);
461        return dialogueChannel.start(dialogue, firstTurn, mDialogueTimeout, dialogueContext);
462    }

464    private C createContext(HttpServletRequest request,
465                            Session<I, O, F, L, C> session,
466                            SynchronousDialogueChannel<I, O, F, L, C> dialogueChannel,

468        C dialogueContext = mDialogueContextFactory.createDialogueContext(request,

482    protected Session<I, O, F, L, C> getSession(HttpServletRequest request) throws SessionNotFoundException {
483        String pathInfo = request.getPathInfo();
484
485        if (pathInfo != null && !pathInfo.equals("/")) {
486            if (pathInfo.startsWith("/")) {
487                pathInfo = pathInfo.substring(1);
488            }
489
490            int firstSlash = pathInfo.indexOf('/');
491            if (firstSlash != -1) {
492                pathInfo = pathInfo.substring(0, firstSlash);
493            }
494
495            return getExistingSession(pathInfo);
496        } else {
497            String sessionId = UUID.randomUUID().toString();
498
499            Session<I, O, F, L, C> session = new Session<I, O, F, L, C>(mSessionContainer, sessionId);
500            mSessionContainer.addSession(session);
501            if (mWebappServerSessionTrackingEnabled) {
502                session.setAssociatedHttpSession(request.getSession());
503            }
504
505            return session;
506        }
507    }
508
509    protected Session<I, O, F, L, C> getExistingSession(String sessionId) throws SessionNotFoundException {
510        Session<I, O, F, L, C> session = mSessionContainer.getSession(sessionId);
511
512        if (session == null) throw new SessionNotFoundException("Unable to find session [" + sessionId + "]");
513
514        return session;
515    }

533    private void commitToResponse(final HttpServletResponse response, ServletResponseContent responseContent)
534            throws IOException {
535        ServletOutputStream outputStream = response.getOutputStream();
536
537        if (mResponseLogger.isDebugEnabled()) {
538            mResponseLogger.debug("Content-length: {}", responseContent.getContentLength());
539            mResponseLogger.debug("Content-type: {}", responseContent.getContentType());
540            mResponseLogger.debug("Content: {}", responseContent.getContentAsString());
541        }
542
543        response.setContentType(responseContent.getContentType());
544        Integer contentLength = responseContent.getContentLength();
545        if (contentLength != null) {
546            response.setContentLength(contentLength);
547        }
548
549        responseContent.writeTo(outputStream);
550    }
