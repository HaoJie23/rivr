005    package com.nuecho.rivr.core.util.logging;
006    
007    import org.slf4j.*;
008    
009    /**
010     * @author Nu Echo Inc.
011     */
012    public class LogUtil {
013    
014        public static void dispatch(Logger log, Marker marker, String message, LogLevel logLevel, Throwable throwable) {
015    
016            boolean hasThrowable = throwable != null;
017    
018            if (marker != null) {
019                switch (logLevel) {
020                    case ALL:
021                    case TRACE:
022                        if (hasThrowable) {
023                            log.trace(marker, message, throwable);
024                        } else {
025                            log.trace(marker, message);
026                        }
027                        break;
028                    case DEBUG:
029                        if (hasThrowable) {
030                            log.debug(marker, message, throwable);
031                        } else {
032                            log.debug(marker, message);
033                        }
034                        break;
035                    case INFO:
036                        if (hasThrowable) {
037                            log.info(marker, message, throwable);
038                        } else {
039                            log.info(marker, message);
040                        }
041                        break;
042                    case WARN:
043                        if (hasThrowable) {
044                            log.warn(marker, message, throwable);
045                        } else {
046                            log.warn(marker, message);
047                        }
048                        break;
049                    case ERROR:
050                        if (hasThrowable) {
051                            log.error(marker, message, throwable);
052                        } else {
053                            log.error(marker, message);
054                        }
055                        break;
056                    case OFF:
057                }
058            } else {
059                switch (logLevel) {
060                    case ALL:
061                    case TRACE:
062                        if (hasThrowable) {
063                            log.trace(message, throwable);
064                        } else {
065                            log.trace(message);
066                        }
067                        break;
068                    case DEBUG:
069                        if (hasThrowable) {
070                            log.debug(message, throwable);
071                        } else {
072                            log.debug(message);
073                        }
074                        break;
075                    case INFO:
076                        if (hasThrowable) {
077                            log.info(message, throwable);
078                        } else {
079                            log.info(message);
080                        }
081                        break;
082                    case WARN:
083                        if (hasThrowable) {
084                            log.warn(message, throwable);
085                        } else {
086                            log.warn(message);
087                        }
088                        break;
089                    case ERROR:
090                        if (hasThrowable) {
091                            log.error(message, throwable);
092                        } else {
093                            log.error(message);
094                        }
095                        break;
096                    case OFF:
097                }
098            }
099        }
100    }
