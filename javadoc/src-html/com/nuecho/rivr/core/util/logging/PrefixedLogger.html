005    package com.nuecho.rivr.core.util.logging;
006    
007    import static org.slf4j.helpers.MessageFormatter.*;
008    
009    import org.slf4j.*;
010    
011    /**
012     * @author Nu Echo Inc.
013     */
014    public class PrefixedLogger extends LoggerAdapter {
015    
016        private final String mPrefix;
017        private final Logger mLogger;
018    
019        public PrefixedLogger(String prefix, Logger logger) {
020            mPrefix = prefix;
021            mLogger = logger;
022        }
023    
024        @Override
025        public String getName() {
026            return mLogger.getName();
027        }
028    
029        @Override
030        public boolean isTraceEnabled() {
031            return mLogger.isTraceEnabled();
032        }
033    
034        @Override
035        public boolean isTraceEnabled(Marker marker) {
036            return mLogger.isTraceEnabled(marker);
037        }
038    
039        @Override
040        public boolean isDebugEnabled() {
041            return mLogger.isDebugEnabled();
042        }
043    
044        @Override
045        public boolean isDebugEnabled(Marker marker) {
046            return mLogger.isDebugEnabled(marker);
047        }
048    
049        @Override
050        public boolean isInfoEnabled() {
051            return mLogger.isInfoEnabled();
052        }
053    
054        @Override
055        public boolean isInfoEnabled(Marker marker) {
056            return mLogger.isInfoEnabled(marker);
057        }
058    
059        @Override
060        public boolean isWarnEnabled() {
061            return mLogger.isWarnEnabled();
062        }
063    
064        @Override
065        public boolean isWarnEnabled(Marker marker) {
066            return mLogger.isWarnEnabled(marker);
067        }
068    
069        @Override
070        public boolean isErrorEnabled() {
071            return mLogger.isErrorEnabled();
072        }
073    
074        @Override
075        public boolean isErrorEnabled(Marker marker) {
076            return mLogger.isErrorEnabled(marker);
077        }
078    
079        @Override
080        protected void log(Marker marker, LogLevel logLevel, String message, Throwable throwable) {
081            LogUtil.dispatch(mLogger, marker, mPrefix + message, logLevel, throwable);
082        }
083    
084        @Override
085        protected void logWithFormatting(Marker marker, LogLevel logLevel, String format, Object... arguments) {
086            LogUtil.dispatch(mLogger, marker, mPrefix + arrayFormat(format, arguments).getMessage(), logLevel, null);
087        }
088    }
