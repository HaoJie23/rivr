001    package com.nuecho.rivr.core.util;
002    
003    public final class ToStringBuilder {
004    
005        private final StringBuilder mBuffer = new StringBuilder();
006        private boolean mEmittedProperty;
007    
008        public ToStringBuilder(Object object) {
009            String className = object.getClass().getName();
010            mBuffer.append(className.substring(className.lastIndexOf('.') + 1));
011            mBuffer.append(" [");
012        }
013    
014        public ToStringBuilder appendItem(String propertyName, Object propertyValue) {
015            if (propertyValue != null) {
016                Class<?> propertyClass = propertyValue.getClass();
017                if (propertyClass.isArray()) {
018                    Class<?> componentType = propertyClass.getComponentType();
019                    if (!componentType.isPrimitive()) {
020                        Object[] array = (Object[]) propertyValue;
021                        insertPropertyName(propertyName, true);
022                        StringUtils.join(mBuffer, array, ", ");
023                        close(true);
024                    } else if (componentType == Boolean.TYPE) {
025                        boolean[] array = (boolean[]) propertyValue;
026                        insertPropertyName(propertyName, true);
027                        StringUtils.join(mBuffer, array, ", ");
028                        close(true);
029                    } else if (componentType == Character.TYPE) {
030                        char[] array = (char[]) propertyValue;
031                        insertPropertyName(propertyName, true);
032                        StringUtils.join(mBuffer, array, ", ");
033                        close(true);
034                    } else if (componentType == Double.TYPE) {
035                        double[] array = (double[]) propertyValue;
036                        insertPropertyName(propertyName, true);
037                        StringUtils.join(mBuffer, array, ", ");
038                        close(true);
039                    } else if (componentType == Float.TYPE) {
040                        float[] array = (float[]) propertyValue;
041                        insertPropertyName(propertyName, true);
042                        StringUtils.join(mBuffer, array, ", ");
043                        close(true);
044                    } else if (componentType == Long.TYPE) {
045                        long[] array = (long[]) propertyValue;
046                        insertPropertyName(propertyName, true);
047                        StringUtils.join(mBuffer, array, ", ");
048                        close(true);
049                    } else if (componentType == Integer.TYPE) {
050                        int[] array = (int[]) propertyValue;
051                        insertPropertyName(propertyName, true);
052                        StringUtils.join(mBuffer, array, ", ");
053                        close(true);
054                    } else if (componentType == Short.TYPE) {
055                        short[] array = (short[]) propertyValue;
056                        insertPropertyName(propertyName, true);
057                        StringUtils.join(mBuffer, array, ", ");
058                        close(true);
059                    } else if (componentType == Byte.TYPE) {
060                        byte[] array = (byte[]) propertyValue;
061                        insertPropertyName(propertyName, true);
062                        StringUtils.join(mBuffer, array, ", ");
063                        close(true);
064                    }
065                } else {
066                    add(propertyName, String.valueOf(propertyValue), false);
067                }
068            } else {
069                add(propertyName, "null", false);
070            }
071            return this;
072        }
073    
074        public ToStringBuilder appendItem(String propertyName, long propertyValue) {
075            add(propertyName, String.valueOf(propertyValue), false);
076            return this;
077        }
078    
079        public ToStringBuilder appendItem(String propertyName, int propertyValue) {
080            add(propertyName, String.valueOf(propertyValue), false);
081            return this;
082        }
083    
084        public ToStringBuilder appendItem(String propertyName, double propertyValue) {
085            add(propertyName, String.valueOf(propertyValue), false);
086            return this;
087        }
088    
089        public ToStringBuilder appendItem(String propertyName, float propertyValue) {
090            add(propertyName, String.valueOf(propertyValue), false);
091            return this;
092        }
093    
094        public ToStringBuilder appendItem(String propertyName, short propertyValue) {
095            add(propertyName, String.valueOf(propertyValue), false);
096            return this;
097        }
098    
099        public ToStringBuilder appendItem(String propertyName, byte propertyValue) {
100            add(propertyName, String.valueOf(propertyValue), false);
101            return this;
102        }
103    
104        public ToStringBuilder appendItem(String propertyName, boolean propertyValue) {
105            add(propertyName, String.valueOf(propertyValue), false);
106            return this;
107        }
108    
109        public ToStringBuilder appendItem(String propertyName, char propertyValue) {
110            add(propertyName, String.valueOf(propertyValue), false);
111            return this;
112        }
113    
114        private void add(String propertyName, String propertyValue, boolean withBraces) {
115            insertPropertyName(propertyName, withBraces);
116            mBuffer.append(propertyValue);
117            close(withBraces);
118        }
119    
120        private void insertPropertyName(String propertyName, boolean withBraces) {
121            if (mEmittedProperty) {
122                mBuffer.append(", ");
123            }
124    
125            if (propertyName.startsWith("m") && propertyName.length() > 1 && Character.isUpperCase(propertyName.charAt(1))) {
126                propertyName = propertyName.substring(1); //strip prefix m
127            }
128    
129            //down-case first letter
130            StringBuffer processedName = new StringBuffer(propertyName);
131            processedName.setCharAt(0, Character.toLowerCase(propertyName.charAt(0)));
132            propertyName = processedName.toString();
133    
134            mBuffer.append(propertyName);
135            mBuffer.append("=");
136    
137            if (withBraces) {
138                mBuffer.append("{");
139            }
140    
141        }
142    
143        private void close(boolean withBraces) {
144            if (withBraces) {
145                mBuffer.append("}");
146            }
147    
148            mEmittedProperty = true;
149        }
150    
151        public String getString() {
152            mBuffer.append("]");
153            return mBuffer.toString();
154        }
155    }
