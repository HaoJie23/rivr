007    import java.util.*;
008    import java.util.concurrent.*;
009    
010    import org.slf4j.*;
011    import org.slf4j.helpers.*;
012    
013    import com.nuecho.rivr.core.channel.*;
014    import com.nuecho.rivr.core.channel.synchronous.step.*;
015    import com.nuecho.rivr.core.dialogue.*;
016    import com.nuecho.rivr.core.servlet.*;

019    /**
020     * @author Nu Echo Inc.
021     */
022    public final class SynchronousDialogueChannel<I extends InputTurn, O extends OutputTurn, F extends FirstTurn, L extends LastTurn, C extends DialogueContext<I, O>>
023            implements DialogueChannel<I, O> {
024    
025        private NamedSynchronousQueue<Step<O, L>> mFromDialogueToController = new NamedSynchronousQueue<Step<O, L>>("dialogue to controller",
026                                                                                                                    true);
027        private NamedSynchronousQueue<I> mFromControllerToDialogue = new NamedSynchronousQueue<I>("controller to dialogue",
028                                                                                                  true);
029        private Thread mDialogueThread;
030    
031        private TimeValue mSendTimeout = TimeValue.seconds(5);
032    
033        private TimeValue mDefaultReceiveFromDialogueTimeout = TimeValue.minutes(1);
034        private TimeValue mDefaultReceiveFromControllerTimeout = TimeValue.minutes(5);
035    
036        private final List<DialogueChannelListener<I, O>> mListener = new ArrayList<DialogueChannelListener<I, O>>();
037        private Logger mLogger = NOPLogger.NOP_LOGGER;
038        private boolean mStopped;
039    
040        private boolean mDialogueStarted;
041        private boolean mDialogueDone;
042    
043        public TimeValue getSendTimeout() {
044            return mSendTimeout;
045        }
046    
047        public void setSendTimeout(TimeValue sendTimeout) {
048            Assert.notNull(sendTimeout, "sendTimeout");
049            mSendTimeout = sendTimeout;
050        }
051    
052        public TimeValue getDefaultReceiveFromDialogueTimeout() {
053            return mDefaultReceiveFromDialogueTimeout;
054        }
055    
056        public void setDefaultReceiveFromDialogueTimeout(TimeValue defaultReceiveFromDialogueTimeout) {
057            Assert.notNull(defaultReceiveFromDialogueTimeout, "defaultReceiveFromDialogueTimeout");
058            mDefaultReceiveFromDialogueTimeout = defaultReceiveFromDialogueTimeout;
059        }
060    
061        public TimeValue getDefaultReceiveFromControllerTimeout() {
062            return mDefaultReceiveFromControllerTimeout;
063        }
064    
065        public void setDefaultReceiveFromControllerTimeout(TimeValue defaultReceiveFromControllerTimeout) {
066            Assert.notNull(defaultReceiveFromControllerTimeout, "defaultReceiveFromControllerTimeout");
067            mDefaultReceiveFromControllerTimeout = defaultReceiveFromControllerTimeout;
068        }
069    
070        public void setLogger(Logger logger) {
071            Assert.notNull(logger, "logger");
072            mLogger = logger;
073        }
074    
075        public Step<O, L> start(final Dialogue<I, O, F, L, C> dialogue,
076                                final F firstTurn,
077                                TimeValue timeout,
078                                final C context) throws Timeout, InterruptedException {
079    
080            Runnable runnable = new Runnable() {
081                @Override
082                public void run() {
083                    mDialogueStarted = true;
084                    for (DialogueChannelListener<I, O> listener : mListener) {
085                        listener.onStart(SynchronousDialogueChannel.this);
086                    }
087    
088                    Step<O, L> lastStep;
089                    try {
090                        L lastTurn = dialogue.run(firstTurn, context);
091                        lastStep = new LastTurnStep<O, L>(lastTurn);
092                    } catch (Throwable throwable) {
093                        mLogger.error("Error in dialogue.", throwable);
094                        lastStep = new ErrorStep<O, L>(throwable);
095                    }
096    
097                    try {
098                        mLogger.trace("Last step: {}", lastStep);
099                        send(mFromDialogueToController, lastStep, mSendTimeout);
100                    } catch (Timeout exception) {
101                        throw new RuntimeException("Timeout while sending final result.", exception);
102                    } catch (InterruptedException exception) {
103                        mLogger.info("Dialogue interrupted while sending final result.");
104                        Thread.currentThread().interrupt();
105                    } finally {
106                        mDialogueDone = true;
107                        mFromDialogueToController = null; // ensure can't receive further output turns from dialogue
108                        mFromControllerToDialogue = null; // ensure can't send further input turns to dialogue
109                        for (DialogueChannelListener<I, O> listener : mListener) {
110                            listener.onStop(SynchronousDialogueChannel.this);
111                        }
112                        mLogger.info("Dialogue ended.");
113                    }
114                }
115            };
116    
117            mDialogueThread = new Thread(runnable, "Dialogue " + context.getDialogueId());
118            mDialogueThread.start();
119            mLogger.info("Dialogue started.");
120    
121            return receive(mFromDialogueToController, timeout);
122        }
123    
124        public boolean isDialogueStarted() {
125            return mDialogueStarted;
126        }
127    
128        public boolean isDialogueDone() {
129            return mDialogueDone;
130        }
131    
132        public boolean isDialogueActive() {
133            return mDialogueStarted && !mDialogueDone;
134        }
135    
136        /**
137         * @param timeout Time to wait for dialogue thread to terminate. A value of
138         *            TimeValue.ZERO (or equivalent) means to wait forever.
139         */
140        public void stop(TimeValue timeout) throws InterruptedException {
141            stop();
142            join(timeout);
143        }
144    
145        public void stop() {
146            mStopped = true;
147            mDialogueThread.interrupt();
148        }
149    
150        public void join(TimeValue timeout) throws InterruptedException {
151            mDialogueThread.join(timeout.getMilliseconds());
152        }
153    
154        @Override
155        public I doTurn(O turn, TimeValue timeout) throws Timeout, InterruptedException {
156            verifyState();
157            mLogger.trace("OutputTurn: {}", turn);
158            OutputTurnStep<O, L> turnStep = new OutputTurnStep<O, L>(turn);
159            if (timeout == null) {
160                timeout = mDefaultReceiveFromControllerTimeout;
161            }
162            return exchange(mFromDialogueToController, mFromControllerToDialogue, turnStep, mSendTimeout, timeout);
163        }
164    
165        public Step<O, L> doTurn(I turn, TimeValue timeout) throws Timeout, InterruptedException {
166            verifyState();
167            mLogger.trace("InputTurn: {}", turn);
168            if (timeout == null) {
169                timeout = mDefaultReceiveFromDialogueTimeout;
170            }
171            return exchange(mFromControllerToDialogue, mFromDialogueToController, turn, mSendTimeout, timeout);
172        }
173    
174        private void verifyState() {
175            if (mDialogueThread == null) throw new IllegalStateException("Dialogue is not set");
176            if (!mDialogueThread.isAlive()) throw new IllegalStateException("Dialogue is not started");
177            if (mStopped) throw new IllegalStateException("Dialogue is stopped");
178        }
179    
180        private <S, R> R exchange(NamedSynchronousQueue<S> sendQueue,
181                                  NamedSynchronousQueue<R> receiveQueue,
182                                  S itemToSend,
183                                  TimeValue sendTimeout,
184                                  TimeValue receiveTimeout) throws Timeout, InterruptedException {
185            try {
186                send(sendQueue, itemToSend, sendTimeout);
187                return receive(receiveQueue, receiveTimeout);
188            } catch (InterruptedException interruptedException) {
189                if (mStopped) throw new DialogueChannelStopped();
190                else throw interruptedException;
191            }
192        }
193    
194        private static <R> R receive(NamedSynchronousQueue<R> receiveQueue, TimeValue timeout) throws Timeout,
195                InterruptedException {
196            if (receiveQueue == null) throw new IllegalStateException("Receive queue is closed.");
197            R result = receiveQueue.poll(timeout.getMilliseconds(), TimeUnit.MILLISECONDS);
198            if (result == null)
199                throw new Timeout("Timed-out in receive() after " + timeout + " in [" + receiveQueue.getName() + "]");
200            return result;
201        }
202    
203        private static <S> void send(NamedSynchronousQueue<S> sendQueue, S itemToSend, TimeValue timeout) throws Timeout,
204                InterruptedException {
205            if (sendQueue == null) throw new IllegalStateException("Send queue is closed.");
206            boolean success = sendQueue.offer(itemToSend, timeout.getMilliseconds(), TimeUnit.MILLISECONDS);
207            if (!success) throw new Timeout("Timed-out in send() after " + timeout + " in [" + sendQueue.getName() + "]");
208        }
209    
210        @Override
211        public void addListener(DialogueChannelListener<I, O> listener) {
212            mListener.add(listener);
213        }
214    
215        @Override
216        public void removeListener(DialogueChannelListener<I, O> listener) {
217            mListener.remove(listener);
218        }
