005package com.nuecho.rivr.core.channel.synchronous.step;
006
007import com.nuecho.rivr.core.channel.*;
008import com.nuecho.rivr.core.util.*;
009
010/**
011 * Step wrapping an {@link OutputTurn}. This kind of step tells that the
012 * dialogue emits an {@link OutputTurn} that should be processed by the
013 * controller and that an {@link InputTurn} must be sent back to dialogue
014 * afterward.
015 * 
016 * @param <O> type of {@link OutputTurn}
017 * @param <L> type of {@link LastTurn}
018 * @author Nu Echo Inc.
019 */
020public final class OutputTurnStep<O extends OutputTurn, L extends LastTurn> implements Step<O, L> {
021
022    private final O mOutputTurn;
023
024    public OutputTurnStep(O outputTurn) {
025        Assert.notNull(outputTurn, "outputTurn");
026        mOutputTurn = outputTurn;
027    }
028
029    public O getOutputTurn() {
030        return mOutputTurn;
031    }
032
033    @Override
034    public String toString() {
035        return "Output Turn: " + mOutputTurn;
036    }
037
038}
