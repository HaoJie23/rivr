005package com.nuecho.rivr.core.dialogue;
006
007import com.nuecho.rivr.core.channel.*;
008import com.nuecho.rivr.core.util.*;
009
010/**
011 * Utility class for dialogue-related operations.
012 * 
013 * @author Nu Echo Inc.
014 */
015public final class DialogueUtils {
016
017    private DialogueUtils() {
018        //utility class: instantiation forbidden
019    }
020
021    /**
022     * Performs turn exchange with default timeout.
023     */
024    public static <I extends InputTurn, O extends OutputTurn> I doTurn(O outputTurn, DialogueContext<I, O> context)
025            throws Timeout, InterruptedException {
026        return context.getDialogueChannel().doTurn(outputTurn, null);
027    }
028
029    /**
030     * Performs turn exchange with specified timeout.
031     */
032    public static <I extends InputTurn, O extends OutputTurn> I doTurn(O outputTurn,
033                                                                       DialogueContext<I, O> context,
034                                                                       Duration timeout) throws Timeout,
035            InterruptedException {
036        return context.getDialogueChannel().doTurn(outputTurn, timeout);
037    }
038
039}
