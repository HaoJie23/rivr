004    package com.nuecho.rivr.core.test;
005    
006    import org.slf4j.*;
007    
008    import com.nuecho.rivr.core.channel.*;
009    import com.nuecho.rivr.core.channel.synchronous.*;
010    import com.nuecho.rivr.core.channel.synchronous.step.*;
011    import com.nuecho.rivr.core.dialogue.*;
012    import com.nuecho.rivr.core.servlet.*;
013    import com.nuecho.rivr.core.util.*;
014    import com.nuecho.rivr.core.util.logging.*;
015    
016    /**
017     * @author Nu Echo Inc.
018     */
019    public abstract class TestDialogueChannel<I extends InputTurn, O extends OutputTurn, F extends FirstTurn, L extends LastTurn, C extends DialogueContext<I, O>>
020            implements DialogueChannel<I, O> {
021    
022        private final SynchronousDialogueChannel<I, O, F, L, C> mChannel;
023        private Step<O, L> mLastStep;
024        private final String mName;
025        private TimeValue mDefaultTimeout;
026    
027        private final AccumulatingLog mAccumulatingLog;
028        private Logger mLogger;
029    
030        public TestDialogueChannel(String name, TimeValue defaultTimeout) {
031            mName = name;
032            mChannel = new SynchronousDialogueChannel<I, O, F, L, C>();
033            mAccumulatingLog = new AccumulatingLog(name);
034            setLogger(mAccumulatingLog);
035            mDefaultTimeout = defaultTimeout;
036        }
037    
038        @Override
039        public I doTurn(O outputTurn, TimeValue timeout) throws Timeout, InterruptedException {
040            return mChannel.doTurn(outputTurn, timeout);
041        }
042    
043        @Override
044        public void addListener(DialogueChannelListener<I, O> listener) {
045            mChannel.addListener(listener);
046        }
047    
048        @Override
049        public void removeListener(DialogueChannelListener<I, O> listener) {
050            mChannel.removeListener(listener);
051        }
052    
053        public void getDefaultTimeout(TimeValue defaultTimeout) {
054            mDefaultTimeout = defaultTimeout;
055        }
056    
057        public void setDefaultTimeout(TimeValue defaultTimeout) {
058            Assert.notNull(defaultTimeout, "defaultTimeout");
059            mDefaultTimeout = defaultTimeout;
060        }
061    
062        public void dispose() {
063    
064            if (!mChannel.isDialogueStarted()) throw new AssertionError("The dialogue hasn't run.");
065    
066            if (mChannel.isDialogueActive()) {
067                stop();
068                mLogger.warn("Test done but dialogue still active.");
069            }
070    
071            mAccumulatingLog.clear();
072        }
073    
074        public final void stop() {
075            if (!mChannel.isDialogueActive()) throw new IllegalStateException("Dialogue not active.");
076            try {
077                mChannel.stop(TimeValue.ZERO);
078            } catch (InterruptedException exception) {
079                mLogger.warn("Interrupted.");
080                Thread.currentThread().interrupt();
081            }
082        }
083    
084        public final Step<O, L> startDialogue(Dialogue<I, O, F, L, C> dialogue, F firstTurn, C dialogueContext) {
085            return startDialogue(dialogue, firstTurn, mDefaultTimeout, dialogueContext);
086        }
087    
088        public final Step<O, L> startDialogue(Dialogue<I, O, F, L, C> dialogue,
089                                              F firstTurn,
090                                              TimeValue timeout,
091                                              C dialogueContext) {
092            if (mChannel.isDialogueActive()) throw new IllegalStateException("Dialogue already started.");
093    
094            try {
095                mLastStep = mChannel.start(dialogue, firstTurn, timeout, dialogueContext);
096                return mLastStep;
097            } catch (Exception exception) {
098                AssertionError assertionError = new AssertionError("Error during start.");
099                assertionError.initCause(exception);
100                throw assertionError;
101            }
102        }
103    
104        public final Step<O, L> processInputTurn(I inputTurn) {
105            return processInputTurn(inputTurn, mDefaultTimeout);
106        }
107    
108        public final Step<O, L> processInputTurn(I inputTurn, TimeValue timeout) {
109    
110            if (!mChannel.isDialogueActive()) throw new IllegalStateException("Dialogue not started.");
111    
112            try {
113                mLastStep = mChannel.doTurn(inputTurn, timeout);
114                return mLastStep;
115            } catch (Exception exception) {
116                AssertionError assertionError = new AssertionError("Error during start.");
117                assertionError.initCause(exception);
118                throw assertionError;
119            }
120        }
121    
122        public Logger getLogger() {
123            return mLogger;
124        }
125    
126        public Step<O, L> getLastStep() {
127            return mLastStep;
128        }
129    
130        public final O getLastStepAsOutputTurn() {
131            if (mLastStep instanceof OutputTurnStep) {
132                OutputTurnStep<O, L> turnStep = (OutputTurnStep<O, L>) mLastStep;
133                return turnStep.getOutputTurn();
134            }
135    
136            throw new AssertionError("Last output turn was not a turn step. Last step=" + mLastStep);
137        }
138    
139        public final Throwable getLastAsError() {
140            if (mLastStep instanceof ErrorStep) {
141                ErrorStep<O, L> errorStep = (ErrorStep<O, L>) mLastStep;
142                return errorStep.getThrowable();
143            }
144    
145            throw new AssertionError("Last output turn was not an error step. Last step=" + mLastStep);
146        }
147    
148        public final SynchronousDialogueChannel<I, O, F, L, C> getChannel() {
149            return mChannel;
150        }
151    
152        public final void dumpLogs() {
153            privateDumpLog(prefixLog(new StandardOutputLogger(LogLevel.ALL)));
154        }
155    
156        public final void dumpLogs(LogLevel logLevel) {
157            privateDumpLog(prefixLog(new StandardOutputLogger(logLevel)));
158        }
159    
160        private void privateDumpLog(Logger consoleLog) {
161            mAccumulatingLog.flushTo(consoleLog);
162            StackTraceElement callerStackTraceElement = getStackTraceElement(3);
163            StackTraceElement currentStackTraceElement = getStackTraceElement(2);
164            consoleLog.info(currentStackTraceElement.getMethodName()
165                            + "() called at "
166                            + callerStackTraceElement.getClassName()
167                            + "."
168                            + callerStackTraceElement.getMethodName()
169                            + "("
170                            + callerStackTraceElement.getFileName()
171                            + ":"
172                            + callerStackTraceElement.getLineNumber()
173                            + ")");
174    
175            setLogger(consoleLog);
176        }
177    
178        private void setLogger(Logger log) {
179            mLogger = log;
180            mChannel.setLogger(mLogger);
181        }
182    
183        private static StackTraceElement getStackTraceElement(int index) {
184            return new Exception().getStackTrace()[index];
185        }
186    
187        private PrefixedLogger prefixLog(Logger log) {
188            return new PrefixedLogger("[" + mName + "] ", log);
189        }
190    
191    }
