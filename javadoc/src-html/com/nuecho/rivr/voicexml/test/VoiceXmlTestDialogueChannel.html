005    package com.nuecho.rivr.voicexml.test;
006    
007    import java.util.*;
008    
009    import javax.json.*;
010    
011    import com.nuecho.rivr.core.channel.*;
012    import com.nuecho.rivr.core.channel.synchronous.step.*;
013    import com.nuecho.rivr.core.test.*;
014    import com.nuecho.rivr.core.util.*;
015    import com.nuecho.rivr.voicexml.rendering.voicexml.*;
016    import com.nuecho.rivr.voicexml.turn.first.*;
017    import com.nuecho.rivr.voicexml.turn.input.*;
018    import com.nuecho.rivr.voicexml.turn.last.*;
019    import com.nuecho.rivr.voicexml.turn.output.*;
020    import com.nuecho.rivr.voicexml.turn.output.interaction.*;
021    import com.nuecho.rivr.voicexml.util.json.*;
022    
023    /**
024     * @author Nu Echo Inc.
025     */
026    public class VoiceXmlTestDialogueChannel
027            extends
028            TestDialogueChannel<VoiceXmlInputTurn, VoiceXmlOutputTurn, VoiceXmlFirstTurn, VoiceXmlLastTurn, VoiceXmlDialogueContext> {
029    
030        private static final String CONFIDENCE_PROPERTY = "confidence";
031        private static final String INTERPRETATION_PROPERTY = "interpretation";
032        private static final String INPUTMODE_PROPERTY = "inputmode";
033        private static final String UTTERANCE_PROPERTY = "utterance";
034    
035        public VoiceXmlTestDialogueChannel(String name, TimeValue defaultTimeout) {
036            super(name, defaultTimeout);
037        }
038    
039        public final InteractionTurn getLastInteractionTurn() {
040            OutputTurn lastOutputTurn = getLastStepAsOutputTurn();
041            if (lastOutputTurn instanceof InteractionTurn) return (InteractionTurn) getLastStepAsOutputTurn();
042            throw new AssertionError("Last output turn was not an interaction. Last step=" + lastOutputTurn);
043        }
044    
045        public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processDtmfRecognition(String dtmfString) {
046            return processDtmfRecognition(dtmfString, JsonValue.NULL, null);
047        }
048    
049        public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processDtmfRecognition(String dtmfString,
050                                                                                       JsonValue interpretation,
051                                                                                       MarkInfo markInfo) {
052            JsonArrayBuilder builder = JsonUtils.createArrayBuilder();
053            JsonUtils.add(builder, createDtmfHypothesis(dtmfString, interpretation));
054            return processRecognition(new RecognitionInfo(builder.build(), markInfo));
055        }
056    
057        public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processRecognition(JsonArray recognitionResult) {
058            return processRecognition(new RecognitionInfo(recognitionResult, null));
059        }
060    
061        public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processRecognition(RecognitionInfo recognitionInfo) {
062            VoiceXmlInputTurn inputTurn = new VoiceXmlInputTurn();
063            inputTurn.setRecognitionInfo(recognitionInfo);
064            return processInputTurn(inputTurn);
065        }
066    
067        public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processScriptExecutionTurn(JsonValue value) {
068            return processValue(value);
069        }
070    
071        public Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processValue(JsonValue value) {
072            VoiceXmlInputTurn inputTurn = new VoiceXmlInputTurn();
073            inputTurn.setJsonValue(value);
074            return processInputTurn(inputTurn);
075        }
076    
077        public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processRecording(RecordingInfo recordingInfo) {
078            VoiceXmlInputTurn inputTurn = new VoiceXmlInputTurn();
079            inputTurn.setRecordingInfo(recordingInfo);
080            return processInputTurn(inputTurn);
081        }
082    
083        public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processNoAction() {
084            return processInputTurn(new VoiceXmlInputTurn());
085        }
086    
087        public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processNoMatch() {
088            return processEvent(VoiceXmlEvent.NO_MATCH);
089        }
090    
091        public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processNoInput() {
092            return processEvent(VoiceXmlEvent.NO_INPUT);
093        }
094    
095        public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processMaxSpeechTimeout() {
096            return processEvent(VoiceXmlEvent.MAX_SPEECH_TIMEOUT);
097        }
098    
099        public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processHangup() {
100            return processEvent(VoiceXmlEvent.CONNECTION_DISCONNECT_HANGUP);
101        }
102    
103        public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processPlatformError() {
104            return processEvent(VoiceXmlEvent.ERROR);
105        }
106    
107        public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processTransferResult(TransferStatusInfo transferStatusInfo) {
108            VoiceXmlInputTurn inputTurn = new VoiceXmlInputTurn();
109            inputTurn.setTransferResult(transferStatusInfo);
110            return processInputTurn(inputTurn);
111        }
112    
113        public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processTransferInvalidDestinationResult() {
114            return processEvent(VoiceXmlEvent.ERROR_CONNECTION_BAD_DESTINATION);
115        }
116    
117        public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processTransferDisconnect() {
118            return processEvent(VoiceXmlEvent.CONNECTION_DISCONNECT_TRANSFER);
119        }
120    
121        public final Step<VoiceXmlOutputTurn, VoiceXmlLastTurn> processEvent(String event) {
122            VoiceXmlInputTurn inputTurn = new VoiceXmlInputTurn();
123            inputTurn.setEvents(createSingleEventList(event));
124            return processInputTurn(inputTurn);
125        }
126    
127        private ArrayList<VoiceXmlEvent> createSingleEventList(String eventName) {
128            VoiceXmlEvent event = new VoiceXmlEvent(eventName, null);
129            return new ArrayList<VoiceXmlEvent>(Collections.singleton(event));
130        }
131    
132        private JsonObject createDtmfHypothesis(String dtmfString, JsonValue interpretation) {
133            return createHypothesis(dtmfString, interpretation, InputMode.DTMF, 1.0);
134        }
135    
136        private JsonObject createHypothesis(String uterance,
137                                            JsonValue interpretation,
138                                            InputMode inputMode,
139                                            double confidence) {
140            JsonObjectBuilder builder = JsonUtils.createObjectBuilder();
141            JsonUtils.add(builder, UTTERANCE_PROPERTY, uterance);
142            JsonUtils.add(builder, INPUTMODE_PROPERTY, inputMode.getKey());
143            JsonUtils.add(builder, INTERPRETATION_PROPERTY, interpretation);
144            builder.add(CONFIDENCE_PROPERTY, confidence);
145            return builder.build();
146        }
147    }
