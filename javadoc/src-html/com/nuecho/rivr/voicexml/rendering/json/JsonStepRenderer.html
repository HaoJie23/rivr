005    package com.nuecho.rivr.voicexml.rendering.json;
006    
007    import java.io.*;
008    
009    import javax.json.*;
010    import javax.servlet.http.*;
011    
012    import com.nuecho.rivr.core.servlet.*;
013    import com.nuecho.rivr.core.util.*;
014    import com.nuecho.rivr.voicexml.rendering.json.JsonServletResponseContent.JsonpMode;
015    import com.nuecho.rivr.voicexml.rendering.voicexml.*;
016    import com.nuecho.rivr.voicexml.turn.input.*;
017    import com.nuecho.rivr.voicexml.turn.last.*;
018    import com.nuecho.rivr.voicexml.turn.output.*;
019    import com.nuecho.rivr.voicexml.util.json.*;
020    
021    /**
022     * @author Nu Echo Inc.
023     */
024    public class JsonStepRenderer implements
025            StepRenderer<VoiceXmlInputTurn, VoiceXmlOutputTurn, VoiceXmlLastTurn, VoiceXmlDialogueContext> {
026    
027        private static final String TEXTAREA_PARAMETER = "textarea";
028        private static final String CALLBACK_PARAMETER = "callback";
029        private static final String TYPE_OUTPUT_TURN = "outputTurn";
030        private static final String TYPE_LAST_TURN = "lastTurn";
031        private static final String TYPE_PROPERTY = "type";
032        private static final String TURN_PROPERTY = "turn";
033        private static final String SERVLET_PATH_PROPERTY = "servletPath";
034        private static final String VOICE_XML_PROPERTY = "voiceXml";
035    
036        private final VoiceXmlStepRenderer mVoiceXmlStepRenderer;
037    
038        public JsonStepRenderer(VoiceXmlStepRenderer voiceXmlStepRenderer) {
039            mVoiceXmlStepRenderer = voiceXmlStepRenderer;
040        }
041    
042        @Override
043        public ServletResponseContent createDocumentForOutputTurn(VoiceXmlOutputTurn outputTurn,
044                                                                  HttpServletRequest request,
045                                                                  HttpServletResponse response,
046                                                                  VoiceXmlDialogueContext dialogueContext)
047                throws StepRendererException {
048            Assert.notNull(outputTurn, "outputTurn");
049            JsonValue turn = outputTurn.asJson();
050    
051            JsonObjectBuilder builder = JsonUtils.createObjectBuilder();
052            JsonUtils.add(builder, TYPE_PROPERTY, TYPE_OUTPUT_TURN);
053            JsonUtils.add(builder, SERVLET_PATH_PROPERTY, VoiceXmlDomUtil.getSubmitPathForTurn(dialogueContext, outputTurn));
054            JsonUtils.add(builder, TURN_PROPERTY, turn);
055    
056            ServletResponseContent voiceXmlResponseContent = mVoiceXmlStepRenderer.createDocumentForOutputTurn(outputTurn,
057                                                                                                               request,
058                                                                                                               response,
059                                                                                                               dialogueContext);
060    
061            return processResponse(request, builder, voiceXmlResponseContent);
062    
063        }
064    
065        @Override
066        public ServletResponseContent createDocumentForLastTurn(VoiceXmlLastTurn lastTurn,
067                                                                HttpServletRequest request,
068                                                                HttpServletResponse response,
069                                                                VoiceXmlDialogueContext dialogueContext)
070                throws StepRendererException {
071            Assert.notNull(lastTurn, "lastTurn");
072            JsonValue turn = lastTurn.asJson();
073    
074            JsonObjectBuilder builder = JsonUtils.createObjectBuilder();
075            JsonUtils.add(builder, TYPE_PROPERTY, TYPE_LAST_TURN);
076            JsonUtils.add(builder, TURN_PROPERTY, turn);
077    
078            ServletResponseContent voiceXmlResponseContent = mVoiceXmlStepRenderer.createDocumentForLastTurn(lastTurn,
079                                                                                                             request,
080                                                                                                             response,
081                                                                                                             dialogueContext);
082    
083            return processResponse(request, builder, voiceXmlResponseContent);
084    
085        }
086    
087        private JsonServletResponseContent processResponse(HttpServletRequest request,
088                                                           JsonObjectBuilder builder,
089                                                           ServletResponseContent voiceXmlServletResponseContent)
090                throws StepRendererException {
091            ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream();
092            try {
093                voiceXmlServletResponseContent.writeTo(byteArrayOutputStream);
094            } catch (IOException exception1) {
095                throw new StepRendererException("Error while wirting VoiceXML document");
096            }
097    
098            try {
099                String voiceXml = new String(byteArrayOutputStream.toByteArray(), "utf-8");
100                JsonUtils.add(builder, VOICE_XML_PROPERTY, voiceXml);
101            } catch (UnsupportedEncodingException exception) {
102                throw new AssertionError("Missing UTF-8 encoding");
103            }
104    
105            String callback = request.getParameter(CALLBACK_PARAMETER);
106            String textarea = request.getParameter(TEXTAREA_PARAMETER);
107    
108            JsonpMode jsonpMode = callback != null
109                    ? textarea != null ? JsonpMode.TEXTAREA : JsonpMode.NORMAL
110                    : JsonpMode.DISABLED;
111            return new JsonServletResponseContent(builder.build(), jsonpMode, callback);
112        }
113    
114    }
