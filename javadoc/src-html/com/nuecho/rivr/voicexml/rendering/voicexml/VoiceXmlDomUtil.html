191            Document document = DomUtils.createDocument(VXML_ELEMENT);

195            vxmlElement.setAttribute(VERSION_ATTRIBUTE, "2.1");

205        public static Document createDocument(VoiceXmlDialogueContext voiceXmlDialogueContext, NamedTurn turn) {
206            Element vxmlElement = createVoiceXmlDocumentRoot(voiceXmlDialogueContext);
207    
208            String rootDocumentPath = voiceXmlDialogueContext.getContextPath()
209                                      + voiceXmlDialogueContext.getServletPath()
210                                      + VoiceXmlDialogueServlet.ROOT_PATH
211                                      + voiceXmlDialogueContext.getDialogueId();
212    
213            vxmlElement.setAttribute(APPLICATION_ATTRIBUTE, rootDocumentPath);
214    
215            StringBuilder script = new StringBuilder();
216            script.append(RIVR_SCOPE_OBJECT)
217                  .append(".")
218                  .append(LOCAL_ERROR_HANDLING_PROPERTY)
219                  .append(" = ")
220                  .append(FALSE)
221                  .append("; ");
222    
223            if (turn != null) {
224                script.append(RIVR_SCOPE_OBJECT)
225                      .append(".")
226                      .append(RIVR_TURN_NAME_PROPERTY)
227                      .append(" = ")
228                      .append(createEcmaScriptStringLiteral(turn.getName()))
229                      .append("; ");
230            }
231    
232            script.append(RIVR_SCOPE_OBJECT)
233                  .append(".")
234                  .append(RIVR_TURN_INDEX_PROPERTY)
235                  .append(" = ")
236                  .append(voiceXmlDialogueContext.getTurnIndex())
237                  .append("; ");
238    
239            script.append(RIVR_INPUT_TURN_SCOPE_OBJECT).append(" = {};");
240            createScript(vxmlElement, script.toString());
241    
242            processFetchRendering(voiceXmlDialogueContext, vxmlElement);
243    
244            processProperties(voiceXmlDialogueContext, vxmlElement);
245    
246            return vxmlElement.getOwnerDocument();
247        }

258        private static void processFetchRendering(VoiceXmlDialogueContext voiceXmlDialogueContext, Element vxmlElement) {
259            FetchConfiguration fetchConfiguration = voiceXmlDialogueContext.getFetchConfiguration();
260            if (fetchConfiguration == null) return;
261    
262            Recording fetchAudio = fetchConfiguration.getDefaultFetchAudio();
263            if (fetchAudio != null) {
264                addFetchAudioProperty(vxmlElement, fetchAudio);
265            }
266    
267            addTimeProperty(vxmlElement, FETCH_TIMEOUT_PROPERTY, fetchConfiguration.getDefaultFetchTimeout());
268    
269            applyDefaultResourceFetchConfiguration(vxmlElement, fetchConfiguration.getDefaultAudioFetchConfiguration());
270            applyDefaultResourceFetchConfiguration(vxmlElement, fetchConfiguration.getDefaultGrammarFetchConfiguration());
271            applyDefaultResourceFetchConfiguration(vxmlElement, fetchConfiguration.getDefaultObjectFetchConfiguration());
272            applyDefaultResourceFetchConfiguration(vxmlElement, fetchConfiguration.getDefaultScriptFetchConfiguration());
273        }

301        public static void createPrompt(String language,
302                                        Element parent,
303                                        VoiceXmlDialogueContext voiceXmlDialogueContext,
304                                        Boolean bargeIn,
305                                        List<? extends AudioItem> audioItems) throws VoiceXmlDocumentRenderingException {
306    
307            Element promptElement = DomUtils.appendNewElement(parent, PROMPT_ATTRIBUTE);
308            setBooleanAttribute(promptElement, BARGE_IN_PROPERTY, bargeIn);
309    
310            String documentLanguage = voiceXmlDialogueContext.getLanguage();
311            if (language != null && !language.equals(documentLanguage)) {
312                promptElement.setAttribute(XML_LANGUAGE_ATTRIBUTE, language);
313            }
314    
315            processAudioItems(audioItems, promptElement);
316        }

318        public static void processAudioItems(List<? extends AudioItem> audioItems, Element promptElement)
319                throws VoiceXmlDocumentRenderingException {
320    
321            boolean lastItemWasText = false;
322            for (AudioItem audioItem : audioItems) {
323                if (audioItem instanceof Recording) {
324                    Recording recording = (Recording) audioItem;
325                    Element audioElement = DomUtils.appendNewElement(promptElement, AUDIO_ELEMENT);
326                    audioElement.setAttribute(SRC_ATTRIBUTE, recording.getPath());
327    
328                    SynthesisText synthesis = recording.getSynthesisText();
329    
330                    if (synthesis != null) {
331                        if (synthesis.isSsml()) {
332                            DocumentFragment ssmlNodes = synthesis.getDocumentFragment();
333                            audioElement.appendChild(audioElement.getOwnerDocument().importNode(ssmlNodes, true));
334                            lastItemWasText = false;
335                        } else {
336                            DomUtils.appendNewText(audioElement, synthesis.getText());
337                            lastItemWasText = true;
338                        }
339                    }
340    
341                    applyRessourceFetchConfiguration(audioElement, recording.getResourceFetchConfiguration());
342                } else if (audioItem instanceof SynthesisText) {
343                    SynthesisText synthesisText = (SynthesisText) audioItem;
344    
345                    if (synthesisText.isSsml()) {
346                        NodeList childNodes = synthesisText.getDocumentFragment().getChildNodes();
347                        Document document = promptElement.getOwnerDocument();
348    
349                        for (int childNodeIndex = 0; childNodeIndex < childNodes.getLength(); childNodeIndex++) {
350                            Node node = document.importNode(childNodes.item(childNodeIndex).cloneNode(true), true);
351                            promptElement.appendChild(node);
352                        }
353                        lastItemWasText = false;
354                    } else {
355                        String spaceIfRequired = lastItemWasText ? " " : "";
356                        DomUtils.appendNewText(promptElement, spaceIfRequired + synthesisText.getText());
357                        lastItemWasText = true;
358                    }
359                } else if (audioItem instanceof Pause) {
360                    Pause pause = (Pause) audioItem;
361                    Element breakElement = DomUtils.appendNewElement(promptElement, BREAK_ELEMENT);
362                    setTimeAttribute(breakElement, TIME_ATTRIBUTE, pause.getDuration());
363                    lastItemWasText = false;
364                } else if (audioItem instanceof Mark) {
365                    Mark mark = (Mark) audioItem;
366                    Element markElement = DomUtils.appendNewElement(promptElement, MARK_ELEMENT);
367                    markElement.setAttribute(NAME_ATTRIBUTE, mark.getName());
368                    lastItemWasText = false;
369                } else if (audioItem instanceof ClientSideRecording) {
370                    ClientSideRecording clientSideRecording = (ClientSideRecording) audioItem;
371                    Element audioElement = DomUtils.appendNewElement(promptElement, AUDIO_ELEMENT);
372                    audioElement.setAttribute(EXPR_ATTRIBUTE, clientSideRecording.getExpression());
373                    lastItemWasText = false;
374                } else throw new VoiceXmlDocumentRenderingException("Cannot handle prompt rendering element of type '"
375                                                                    + audioItem.getClass()
376                                                                    + "'");
377            }
378        }

454        public static String createEcmaScriptStringLiteral(String content) {
455            if (content == null) return "undefined";
456    
457            //believe it: this creates \\ for each \
458            content = ECMASCRIPT_BACKSLASH_PATTERN.matcher(content).replaceAll("\\\\\\\\");
459    
460            content = ECMASCRIPT_QUOTE_PATTERN.matcher(content).replaceAll("\\\\'");
461            content = ECMASCRIPT_LINEFEED_PATTERN.matcher(content).replaceAll("\\u000a");
462            content = ECMASCRIPT_CARRIAGE_RETURN_PATTERN.matcher(content).replaceAll("\\u000d");
463            content = ECMASCRIPT_LINE_SEPARATOR_PATTERN.matcher(content).replaceAll("\\u2028");
464            content = ECMASCRIPT_PARAGRAPH_SEPARATOR_PATTERN.matcher(content).replaceAll("\\u2029");
465    
466            StringBuilder out = new StringBuilder();
467            out.append("'").append(content).append("'");
468            return out.toString();
469        }

482                                                   SubmitMethod method,

485            Element submitElement = DomUtils.appendNewElement(parent, SUBMIT_ELEMENT);

493            return submitElement;
494        }

536        public static void applyFetchAudio(Element element, Recording fetchAudio) {
537            if (fetchAudio != null) {

661        public static void renderGrammars(List<? extends GrammarItem> grammarItems, Element parent, String mode)
662                throws VoiceXmlDocumentRenderingException {
663            for (GrammarItem grammarItem : grammarItems) {
664                Element grammarElement;
665    
666                if (grammarItem instanceof GrammarReference) {
667                    grammarElement = DomUtils.appendNewElement(parent, GRAMMAR_ELEMENT);
668                    GrammarReference grammarReference = (GrammarReference) grammarItem;
669    
670                    VoiceXmlDomUtil.renderGrammarReference(grammarReference, grammarElement);
671                } else if (grammarItem instanceof InlineStringGrammar) {
672                    grammarElement = DomUtils.appendNewElement(parent, GRAMMAR_ELEMENT);
673                    InlineStringGrammar inlineStringGrammar = (InlineStringGrammar) grammarItem;
674                    VoiceXmlDomUtil.renderInlineStringGrammar(grammarElement, inlineStringGrammar);
675                } else if (grammarItem instanceof InlineXmlGrammar) {
676                    InlineXmlGrammar inlineXmlGrammar = (InlineXmlGrammar) grammarItem;
677                    Document document = inlineXmlGrammar.getDocument();
678                    Element documentRootElement = document.getDocumentElement();
679                    if (!documentRootElement.getTagName().equals(GRAMMAR_ELEMENT))
680                        throw new VoiceXmlDocumentRenderingException("XML grammar root element must be 'gramar'");
681    
682                    grammarElement = (Element) parent.getOwnerDocument().importNode(documentRootElement, true);
683                    parent.appendChild(grammarElement);
684                } else throw new AssertionError("Unsupported grammar item: " + grammarItem);
685    
686                Double weight = grammarItem.getWeight();
687                if (weight != null) {
688                    DecimalFormat weigthDecimalFormat = new DecimalFormat("#.##########");
689                    double weightDoubleValue = weight.doubleValue();
690                    grammarElement.setAttribute(WEIGHT_ATTRIBUTE, weigthDecimalFormat.format(weightDoubleValue));
691                }
692    
693                setAttribute(grammarElement, TYPE_ATTRIBUTE, grammarItem.getMediaType());
694                setAttribute(grammarElement, MODE_ATTRIBUTE, mode);
695            }
696        }

705        }
