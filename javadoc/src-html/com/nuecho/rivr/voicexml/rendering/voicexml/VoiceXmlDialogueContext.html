005    package com.nuecho.rivr.voicexml.rendering.voicexml;
006    
007    import java.util.*;
008    
009    import org.slf4j.*;
010    
011    import com.nuecho.rivr.core.channel.*;
012    import com.nuecho.rivr.core.dialogue.*;
013    import com.nuecho.rivr.core.util.*;
014    import com.nuecho.rivr.voicexml.rendering.voicexml.errorhandling.*;
015    import com.nuecho.rivr.voicexml.turn.input.*;
016    import com.nuecho.rivr.voicexml.turn.output.*;
017    import com.nuecho.rivr.voicexml.turn.output.fetch.*;
018    
019    /**
020     * @author Nu Echo Inc.
021     */
022    public final class VoiceXmlDialogueContext implements DialogueContext<VoiceXmlInputTurn, VoiceXmlOutputTurn> {
023    
024        private String mLanguage;
025    
026        private FetchConfiguration mFetchConfiguration;
027        private final Map<String, String> mProperties = new HashMap<String, String>();
028    
029        private String mDialogueId;
030    
031        private final String mContextPath;
032        private final String mServletPath;
033        private int mTurnIndex;
034    
035        private final DialogueChannel<VoiceXmlInputTurn, VoiceXmlOutputTurn> mDialogueChannel;
036    
037        private final Logger mLogger;
038    
039        private FatalErrorFormFactory mFatalErrorFormFactory = new ExitFatalErrorFormFactory();
040    
041        public VoiceXmlDialogueContext(DialogueChannel<VoiceXmlInputTurn, VoiceXmlOutputTurn> dialogueChannel,
042                                       Logger logger,
043                                       String dialogueId,
044                                       String contextPath,
045                                       String servletPath) {
046            mDialogueChannel = dialogueChannel;
047            mLogger = logger;
048            mDialogueId = dialogueId;
049            mContextPath = contextPath;
050            mServletPath = servletPath;
051        }
052    
053        public void setDialogueId(String dialogueId) {
054            mDialogueId = dialogueId;
055        }
056    
057        @Override
058        public String getDialogueId() {
059            return mDialogueId;
060        }
061    
062        public String getContextPath() {
063            return mContextPath;
064        }
065    
066        public String getServletPath() {
067            return mServletPath;
068        }
069    
070        public String getLanguage() {
071            return mLanguage;
072        }
073    
074        public void setLanguage(String language) {
075            mLanguage = language;
076        }
077    
078        public FetchConfiguration getFetchConfiguration() {
079            return mFetchConfiguration;
080        }
081    
082        public void setFetchConfiguration(FetchConfiguration fetchConfiguration) {
083            mFetchConfiguration = fetchConfiguration;
084        }
085    
086        public Map<String, String> getProperties() {
087            return mProperties;
088        }
089    
090        public void incrementTurnIndex() {
091            mTurnIndex++;
092        }
093    
094        public int getTurnIndex() {
095            return mTurnIndex;
096        }
097    
098        @Override
099        public DialogueChannel<VoiceXmlInputTurn, VoiceXmlOutputTurn> getDialogueChannel() {
100            return mDialogueChannel;
101        }
102    
103        @Override
104        public Logger getLogger() {
105            return mLogger;
106        }
107    
108        public void setFatalErrorFormFactory(FatalErrorFormFactory fatalErrorFormFactory) {
109            mFatalErrorFormFactory = fatalErrorFormFactory;
110        }
111    
112        public FatalErrorFormFactory getFatalErrorFormFactory() {
113            return mFatalErrorFormFactory;
114        }
115    
116        @Override
117        public String toString() {
118            ToStringBuilder builder = new ToStringBuilder(this);
119            builder.appendItem("mLanguage", mLanguage);
120            builder.appendItem("mFetchConfiguration", mFetchConfiguration);
121            builder.appendItem("mProperties", mProperties);
122            builder.appendItem("mDialogueId", mDialogueId);
123            builder.appendItem("mContextPath", mContextPath);
124            builder.appendItem("mServletPath", mServletPath);
125            builder.appendItem("mTurnIndex", mTurnIndex);
126            builder.appendItem("mFatalErrorFormFactory", mFatalErrorFormFactory);
127            return builder.getString();
128        }
129    
130    }
