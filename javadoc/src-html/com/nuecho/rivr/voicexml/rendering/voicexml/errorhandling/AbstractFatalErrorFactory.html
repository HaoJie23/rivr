005    package com.nuecho.rivr.voicexml.rendering.voicexml.errorhandling;
006    
007    import org.w3c.dom.*;
008    
009    import com.nuecho.rivr.core.util.*;
010    import com.nuecho.rivr.voicexml.rendering.voicexml.*;
011    import com.nuecho.rivr.voicexml.turn.*;
012    
013    /**
014     * @author Nu Echo Inc.
015     */
016    public class AbstractFatalErrorFactory implements FatalErrorFormFactory {
017    
018        private final String mElementName;
019    
020        public AbstractFatalErrorFactory(String elementName) {
021            mElementName = elementName;
022        }
023    
024        @Override
025        public void addFatalErrorForm(Element form, VoiceXmlDocumentTurn turn) {
026            Element blockElement = VoiceXmlDomUtil.addBlockElement(form);
027            Element element = DomUtils.appendNewElement(blockElement, mElementName);
028            customizeForm(form, blockElement, element);
029        }
030    
031        /**
032         * @param form
033         * @param blockElement
034         * @param element
035         */
036        protected void customizeForm(Element form, Element blockElement, Element element) {}
037    
038    }
