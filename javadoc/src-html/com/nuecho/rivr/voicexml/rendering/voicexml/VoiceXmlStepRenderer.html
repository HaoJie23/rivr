005    package com.nuecho.rivr.voicexml.rendering.voicexml;
006    
007    import java.util.*;
008    
009    import javax.servlet.http.*;
010    
011    import org.w3c.dom.*;
012    
013    import com.nuecho.rivr.core.servlet.*;
014    import com.nuecho.rivr.core.util.*;
015    import com.nuecho.rivr.voicexml.turn.*;
016    import com.nuecho.rivr.voicexml.turn.input.*;
017    import com.nuecho.rivr.voicexml.turn.last.*;
018    import com.nuecho.rivr.voicexml.turn.output.*;
019    
020    /**
021     * @author Nu Echo Inc.
022     */
023    public class VoiceXmlStepRenderer implements
024            StepRenderer<VoiceXmlInputTurn, VoiceXmlOutputTurn, VoiceXmlLastTurn, VoiceXmlDialogueContext> {
025    
026        private static final String VOICE_XML_MIME_TYPE = "application/voicexml+xml";
027    
028        private final List<VoiceXmlDocumentAdapter> mVoiceXmlDocumentAdapters;
029    
030        public VoiceXmlStepRenderer(List<? extends VoiceXmlDocumentAdapter> voiceXmlDocumentAdapters) {
031            mVoiceXmlDocumentAdapters = new ArrayList<VoiceXmlDocumentAdapter>(voiceXmlDocumentAdapters);
032        }
033    
034        @Override
035        public ServletResponseContent createDocumentForOutputTurn(VoiceXmlOutputTurn outputTurn,
036                                                                  HttpServletRequest request,
037                                                                  HttpServletResponse response,
038                                                                  VoiceXmlDialogueContext voiceXmlDialogueContext)
039                throws StepRendererException {
040            return renderTurn(outputTurn, voiceXmlDialogueContext);
041        }
042    
043        @Override
044        public ServletResponseContent createDocumentForLastTurn(VoiceXmlLastTurn lastTurn,
045                                                                HttpServletRequest request,
046                                                                HttpServletResponse response,
047                                                                VoiceXmlDialogueContext voiceXmlDialogueContext)
048                throws StepRendererException {
049            return renderTurn(lastTurn, voiceXmlDialogueContext);
050        }
051    
052        private ServletResponseContent renderTurn(VoiceXmlDocumentTurn turn, VoiceXmlDialogueContext voiceXmlDialogueContext)
053                throws StepRendererException {
054            Assert.notNull(turn, "turn");
055            Document voiceXmlDocument;
056            try {
057                voiceXmlDocument = turn.getVoiceXmlDocument(voiceXmlDialogueContext);
058            } catch (VoiceXmlDocumentRenderingException exception) {
059                throw new StepRendererException(exception);
060            }
061    
062            try {
063                if (mVoiceXmlDocumentAdapters != null) {
064                    for (VoiceXmlDocumentAdapter adapter : mVoiceXmlDocumentAdapters) {
065                        adapter.adaptVoiceXmlDocument(voiceXmlDocument);
066                    }
067                }
068            } catch (VoiceXmlDocumentRenderingException exception) {
069                throw new StepRendererException("Error white applying adapter.", exception);
070            }
071    
072            voiceXmlDialogueContext.incrementTurnIndex();
073            return new XmlDocumentServletResponseContent(voiceXmlDocument, VOICE_XML_MIME_TYPE);
074        }
075    
076    }
