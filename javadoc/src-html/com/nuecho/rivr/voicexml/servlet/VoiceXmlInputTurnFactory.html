067        @Override
068        public VoiceXmlFirstTurn createFirstTurn(HttpServletRequest request, HttpServletResponse response)
069                throws InputTurnFactoryException {
070    
071            Map<String, String> parameters = new HashMap<String, String>();
072    
073            @SuppressWarnings("unchecked")
074            Enumeration<String> parameterNames = request.getParameterNames();
075            while (parameterNames.hasMoreElements()) {
076                String parameterName = parameterNames.nextElement();
077                parameters.put(parameterName, request.getParameter(parameterName));
078            }
079    
080            return new VoiceXmlFirstTurn(parameters);
081        }
082    
083        @Override
084        public VoiceXmlInputTurn createInputTurn(HttpServletRequest request, HttpServletResponse response)
085                throws InputTurnFactoryException {
086            Map<String, Object> parameters = buildParameterMap(request);
087    
088            String result = (String) parameters.get(INPUT_TURN_PARAMETER);
089    
090            if (result == null)
091                throw new InputTurnFactoryException("Unable to process request. Missing 'result' parameter.");
092    
093            JsonReader jsonReader = JsonUtils.createReader(result);
094            JsonObject resultObject = jsonReader.readObject();
095            VoiceXmlInputTurn voiceXmlInputTurn = new VoiceXmlInputTurn();
096    
097            addEvents(resultObject, voiceXmlInputTurn);
098            addObject(resultObject, voiceXmlInputTurn);
099            addTransferStatusInfo(resultObject, voiceXmlInputTurn);
100            addRecognitionInfo(resultObject, voiceXmlInputTurn);
101            addRecordingInfo(resultObject, voiceXmlInputTurn, parameters);
102            return voiceXmlInputTurn;
103        }
104    
105        private static void addEvents(JsonObject resultObject, VoiceXmlInputTurn voiceXmlInputTurn) {
106            if (!resultObject.containsKey(EVENTS_PROPERTY)) return;
107    
108            List<JsonValue> eventObjects = resultObject.getJsonArray(EVENTS_PROPERTY);
109            List<VoiceXmlEvent> events = new ArrayList<VoiceXmlEvent>();
110            for (JsonValue jsonValue : eventObjects) {
111                JsonObject eventObject = (JsonObject) jsonValue;
112    
113                String name = eventObject.getString(EVENT_NAME_PROPERTY);
114                String message = eventObject.getString(EVENT_MESSAGE_PROPERTY, null);
115    
116                events.add(new VoiceXmlEvent(name, message));
117            }
118    
119            voiceXmlInputTurn.setEvents(events);
120        }
121    
122        private static void addObject(JsonObject resultObject, VoiceXmlInputTurn voiceXmlInputTurn) {
123            if (!resultObject.containsKey(VALUE_PROPERTY)) return;
124    
125            JsonValue subdialogueResultJsonValue = resultObject.get(VALUE_PROPERTY);
126            voiceXmlInputTurn.setJsonValue(subdialogueResultJsonValue);
127        }
128    
129        private static void addTransferStatusInfo(JsonObject resultObject, VoiceXmlInputTurn voiceXmlInputTurn) {
130            if (!resultObject.containsKey(TRANSFER_PROPERTY)) return;
131    
132            JsonObject transferResultObject = resultObject.getJsonObject(TRANSFER_PROPERTY);
133    
134            TransferStatus transferStatus = new TransferStatus(transferResultObject.getString(TRANSFER_STATUS_PROPERTY));
135    
136            long durationValue;
137    
138            if (transferResultObject.containsKey(TRANSFER_DURATION_PROPERTY)) {
139                durationValue = JsonUtils.getLongProperty(transferResultObject, TRANSFER_DURATION_PROPERTY);
140            } else {
141                durationValue = 0;
142            }
143    
144            TimeValue duration = TimeValue.milliseconds(durationValue);
145    
146            voiceXmlInputTurn.setTransferResult(new TransferStatusInfo(transferStatus, duration));
147        }
148    
149        private static void addRecordingInfo(JsonObject resultObject,
150                                             VoiceXmlInputTurn voiceXmlInputTurn,
151                                             Map<String, Object> parameters) {
152            if (!resultObject.containsKey(RECORDING_META_DATA_PROPERTY)) return;
153    
154            JsonObject recordingMetaData = resultObject.getJsonObject(RECORDING_META_DATA_PROPERTY);
155    
156            TimeValue duration;
157            if (recordingMetaData.containsKey(DURATION_PROPERTY)) {
158                long durationInMilliseconds = JsonUtils.getLongProperty(recordingMetaData, DURATION_PROPERTY);
159                duration = TimeValue.milliseconds(durationInMilliseconds);
160            } else {
161                duration = null;
162            }
163    
164            boolean maxTime = recordingMetaData.getBoolean(MAX_TIME_PROPERTY, false);
165    
166            String dtmfTermChar = recordingMetaData.getString(TERM_CHAR_PROPERTY, null);
167    
168            RecordingData recordingData;
169            if (!parameters.containsKey(RECORDING_PARAMETER)) {
170                recordingData = null;
171            } else {
172                FileUpload fileUpload = (FileUpload) parameters.get(RECORDING_PARAMETER);
173                recordingData = new RecordingData(fileUpload.getData(), fileUpload.getContentType(), fileUpload.getName());
174            }
175    
176            voiceXmlInputTurn.setRecordingInfo(new RecordingInfo(recordingData, duration, maxTime, dtmfTermChar));
177        }
178    
179        private void addRecognitionInfo(JsonObject jsonObject, VoiceXmlInputTurn voiceXmlInputTurn) {
180    
181            if (!jsonObject.containsKey(RECOGNITION_PROPERTY)) return;
182    
183            JsonObject recognitionObject = jsonObject.getJsonObject(RECOGNITION_PROPERTY);
184    
185            JsonArray recognitionResultArray = recognitionObject.getJsonArray(RESULT_PROPERTY);
186    
187            MarkInfo markInfo = null;
188            if (recognitionObject.containsKey(MARK_PROPERTY)) {
189                JsonObject markObject = recognitionObject.getJsonObject(MARK_PROPERTY);
190                markInfo = new MarkInfo(markObject.getString(MARK_NAME_PROPERTY),
191                                        JsonUtils.getLongProperty(markObject, MARK_TIME_PROPERTY));
192            }
193    
194            voiceXmlInputTurn.setRecognitionInfo(new RecognitionInfo(recognitionResultArray, markInfo));
195        }

197        private Map<String, Object> buildParameterMap(HttpServletRequest request) throws InputTurnFactoryException {
198            Map<String, Object> parameters = new HashMap<String, Object>();
199            if (ServletFileUpload.isMultipartContent(request)) {
200                ServletFileUpload servletFileUpload = new ServletFileUpload();
201                try {
202                    FileItemIterator itemIterator = servletFileUpload.getItemIterator(request);
203                    while (itemIterator.hasNext()) {
204                        FileItemStream fileItemStream = itemIterator.next();
205    
206                        byte[] bytes;
207                        try {
208                            InputStream stream = fileItemStream.openStream();
209                            bytes = IOUtils.toByteArray(stream);
210                        } catch (IOException exception) {
211                            throw new ServletException("Unable to read stream from " + fileItemStream.getFieldName(),
212                                                       exception);
213                        }
214    
215                        if (!fileItemStream.isFormField()) {
216                            FileUpload fileUpload = new FileUpload(fileItemStream.getName(),
217                                                                   fileItemStream.getContentType(),
218                                                                   bytes,
219                                                                   getHeaders(fileItemStream));
220    
221                            parameters.put(fileItemStream.getFieldName(), fileUpload);
222                        } else {
223                            String encoding = findEncoding(request, fileItemStream);
224                            parameters.put(fileItemStream.getFieldName(), new String(bytes, encoding));
225                        }
226    
227                    }
228                } catch (Exception exception) {
229                    throw new InputTurnFactoryException("Unable to get recording.", exception);
230                }
231            } else {
232                @SuppressWarnings("unchecked")
233                Enumeration<String> parameterNames = request.getParameterNames();
234                while (parameterNames.hasMoreElements()) {
235                    String parameterName = parameterNames.nextElement();
236                    parameters.put(parameterName, request.getParameter(parameterName));
237                }
238            }
239    
240            return parameters;
241        }
242    
243        private String findEncoding(HttpServletRequest request, FileItemStream fileItemStream) {
244    
245            String encoding = null;
246    
247            String contentType = fileItemStream.getContentType();
248    
249            if (contentType != null && contentType.startsWith("text/plain")) {
250                Matcher matcher = CHAR_SET_PATTERN.matcher(contentType);
251                if (matcher.find()) {
252                    encoding = matcher.group(1);
253                } else {
254                    encoding = Encoding.US_ASCII.getId();
255                }
256            }
257    
258            if (encoding == null) {
259                encoding = request.getCharacterEncoding();
260            }
261    
262            if (encoding == null) {
263                encoding = Encoding.US_ASCII.getId();
264            }
265    
266            return encoding;
267        }
268    
269        private Map<String, String> getHeaders(FileItemStream fileItemStream) {
270            Map<String, String> headers = new HashMap<String, String>();
271            FileItemHeaders sourceHeaders = fileItemStream.getHeaders();
272    
273            if (sourceHeaders == null) return headers;
274    
275            @SuppressWarnings("unchecked")
276            Iterator<String> headerNames = sourceHeaders.getHeaderNames();
277            while (headerNames.hasNext()) {
278                String headerName = headerNames.next();
279                headers.put(headerName, sourceHeaders.getHeader(headerName));
280            }
281            return headers;
282        }
