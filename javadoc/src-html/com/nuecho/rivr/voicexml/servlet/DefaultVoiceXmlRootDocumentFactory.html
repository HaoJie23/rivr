021    /**
022     * @author Nu Echo Inc.
023     */
024    public class DefaultVoiceXmlRootDocumentFactory implements VoiceXmlRootDocumentFactory {
025    
026        @Override
027        public Document getDocument(HttpServletRequest request,
028                                    Session<VoiceXmlInputTurn, VoiceXmlOutputTurn, VoiceXmlFirstTurn, VoiceXmlLastTurn, VoiceXmlDialogueContext> session) {
029    
030            VoiceXmlDialogueContext dialogueContext = session.getDialogueContext();
031            String contextPath = dialogueContext.getContextPath();
032            String servletPath = dialogueContext.getServletPath();
033            String sessionId = session.getId();
034            return createElement(contextPath, servletPath, sessionId, dialogueContext);
035        }
036    
037        protected Document createElement(String contextPath,
038                                         String servletPath,
039                                         String sessionId,
040                                         VoiceXmlDialogueContext dialogueContext) {
041            Element vxmlElement = createVoiceXmlDocumentRoot(dialogueContext);
042            createVarElement(vxmlElement, RIVR_VARIABLE, "({\"" + RIVR_DIALOGUE_ID_PROPERTY + "\": \"" + sessionId + "\"})");
043    
044            addScript(vxmlElement, VoiceXmlDialogueServlet.RIVR_SCRIPT, contextPath + servletPath);
045            return vxmlElement.getOwnerDocument();
046        }
047    
048        private static void addScript(Element parent, String scriptPath, String contextPath) {
049            Element scriptElement = DomUtils.appendNewElement(parent, SCRIPT_ELEMENT);
050            scriptElement.setAttribute(SRC_ATTRIBUTE, contextPath + scriptPath);
051        }
052    
053    }
