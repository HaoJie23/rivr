072        @Override
073        protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
074            String pathInfo = request.getPathInfo();
075    
076            if (pathInfo != null) {
077                if (pathInfo.startsWith(ROOT_PATH)) {
078                    processRootDocument(request, response, pathInfo);
079                    return;
080                }
081    
082                if (RIVR_SCRIPT.equals(pathInfo)) {
083                    processRessource(request, response, pathInfo);
084                    return;
085                }
086            }
087    
088            super.doGet(request, response);
089        }

112        private void initializeProperties() throws DialogueServletInitializationException {
113    
114            ILoggerFactory loggerFactory = find(INITIAL_ARGUMENT_LOGGER_FACTORY, ILoggerFactory.class);
115            if (loggerFactory != null) {
116                setLoggerFactory(loggerFactory);
117            }
118    
119            VoiceXmlDialogueFactory dialogueFactory = getDialogueFactory();
120            if (dialogueFactory != null) {
121                setDialogueFactory(dialogueFactory);
122            } else {
123                setImplicitDialogueFactory();
124            }
125    
126            VoiceXmlErrorHandler errorHandler = find(INITIAL_ARGUMENT_ERROR_HANDLER, VoiceXmlErrorHandler.class);
127            if (errorHandler != null) {
128                setErrorHandler(errorHandler);
129            }
130    
131            TimeValue sessionScanPeriod = getTimeValue(INITIAL_ARGUMENT_SESSION_SCAN_PERIOD);
132            if (sessionScanPeriod != null) {
133                setSessionScanPeriod(sessionScanPeriod);
134            }
135    
136            TimeValue sessionTimeout = getTimeValue(INITIAL_ARGUMENT_SESSION_TIMEOUT);
137            if (sessionTimeout != null) {
138                setSessionTimeout(sessionTimeout);
139            }
140    
141            TimeValue dialogueTimeout = getTimeValue(INITIAL_ARGUMENT_DIALOGUE_TIMEOUT);
142            if (dialogueTimeout != null) {
143                setDialogueTimeout(dialogueTimeout);
144            }
145        }

147        private void setImplicitDialogueFactory() throws DialogueServletInitializationException {
148            String dialogueClassName = getServletConfig().getInitParameter(INITIAL_ARGUMENT_DIALOGUE_CLASS);
149            if (dialogueClassName == null) return;
150    
151            try {
152                Class<?> rawDialogueClass = Class.forName(dialogueClassName);
153                if (!VoiceXmlDialogue.class.isAssignableFrom(rawDialogueClass)) {
154                    String message = "Dialogue class "
155                                     + rawDialogueClass
156                                     + " does not implements "
157                                     + VoiceXmlDialogue.class.getName()
158                                     + ".";
159                    throw new DialogueServletInitializationException(message);
160                }
161    
162                @SuppressWarnings("unchecked")
163                Class<? extends VoiceXmlDialogue> dialogueClass = (Class<? extends VoiceXmlDialogue>) rawDialogueClass;
164                setDialogueFactory(new SimpleVoiceXmlDialogueFactory(dialogueClass));
165            } catch (ClassNotFoundException exception) {
166                throw new DialogueServletInitializationException("Cannot find dialogue class.", exception);
167            } catch (DialogueFactoryException exception) {
168                throw new DialogueServletInitializationException("Cannot initialize dialogue factory.", exception);
169            }
170        }

172        private TimeValue getTimeValue(String key) throws DialogueServletInitializationException {
173            ServletConfig servletConfig = getServletConfig();
174            String timeValue = servletConfig.getInitParameter(key);
175            if (timeValue == null) return null;
176            try {
177                return TimeValue.parse(timeValue);
178            } catch (IllegalArgumentException exception) {
179                throw new DialogueServletInitializationException("Unable to parse time value.", exception);
180            }
181        }
182    
183        private <T> T find(String prefix, Class<T> type) throws DialogueServletInitializationException {
184            T object = null;
185            ServletConfig servletConfig = getServletConfig();
186            String className = servletConfig.getInitParameter(prefix + ".class");
187            if (className != null) {
188                object = instantiate(className, type, prefix);
189            }
190    
191            String key = servletConfig.getInitParameter(prefix + ".key");
192            if (key != null) {
193                object = findInServletContext(key, type, prefix);
194            }
195            return object;
196        }

198        protected VoiceXmlDialogueFactory getDialogueFactory() throws DialogueServletInitializationException {
199            return find(INITIAL_ARGUMENT_DIALOGUE_FACTORY, VoiceXmlDialogueFactory.class);
200        }

202        private <T> T findInServletContext(String servletContextKey, Class<T> type, String item)
203                throws DialogueServletInitializationException {
204            Object object = getServletContext().getAttribute(servletContextKey);
205    
206            if (object == null)
207                throw new DialogueServletInitializationException("Cannot find "
208                                                                 + item
209                                                                 + " with name ["
210                                                                 + servletContextKey
211                                                                 + "] in servlet context.");
212    
213            if (!(type.isInstance(object)))
214                throw new DialogueServletInitializationException("Servlet context object ["
215                                                                 + servletContextKey
216                                                                 + "] does not implements "
217                                                                 + type.getName()
218                                                                 + ". Actual class is "
219                                                                 + object.getClass().getName());
220            return type.cast(object);
221        }
222    
223        private <T> T instantiate(String className, Class<T> type, String item)
224                throws DialogueServletInitializationException {
225            try {
226                Class<?> classObject = Class.forName(className);
227                if (!type.isAssignableFrom(classObject))
228                    throw new DialogueServletInitializationException("Incompatible class type.");
229                return type.cast(classObject.newInstance());
230            } catch (ClassNotFoundException exception) {
231                throw new DialogueServletInitializationException("Cannot find " + item + " class '" + className + "'",
232                                                                 exception);
233            } catch (InstantiationException exception) {
234                throw new DialogueServletInitializationException("Cannot instantiate "
235                                                                 + item
236                                                                 + " of class '"
237                                                                 + className
238                                                                 + "'", exception);
239            } catch (IllegalAccessException exception) {
240                throw new DialogueServletInitializationException("Cannot access " + item + " class '" + className + "'",
241                                                                 exception);
242            }
243        }

245        @Override
246        protected StepRenderer<VoiceXmlInputTurn, VoiceXmlOutputTurn, VoiceXmlLastTurn, VoiceXmlDialogueContext> getStepRenderer(HttpServletRequest request,
247                                                                                                                                 Session<VoiceXmlInputTurn, VoiceXmlOutputTurn, VoiceXmlFirstTurn, VoiceXmlLastTurn, VoiceXmlDialogueContext> session) {
248            String acceptHeader = request.getHeader(ACCEPT_HEADER);
249            if (acceptHeader == null) return mVoiceXmlStepRenderer;
250            if (acceptHeader.indexOf("application/json") != -1) return mJsonStepRenderer;
251            if (acceptHeader.indexOf("application/javascript") != -1) return mJsonStepRenderer;
252            if (acceptHeader.indexOf("application/voicexml+xml") != -1) return mVoiceXmlStepRenderer;
253            return mVoiceXmlStepRenderer;
254        }

256        private void processRessource(HttpServletRequest request, HttpServletResponse response, String pathInfo)
257                throws IOException, ServletException {
258            InputStream inputStream = VoiceXmlDialogueServlet.class.getResourceAsStream(pathInfo.substring(1));
259            byte[] bytes = IOUtils.toByteArray(inputStream);
260            String eTag = getETag(bytes);
261            String ifNoneMatch = request.getHeader(IF_NONE_MATCH);
262    
263            if (eTag.equals(ifNoneMatch)) {
264                response.sendError(HttpServletResponse.SC_NOT_MODIFIED, "Not Modified");
265            } else {
266                response.setContentType(JAVASCRIPT_CONTENT_TYPE);
267                response.addHeader(ETAG, eTag);
268                response.getOutputStream().write(bytes);
269            }
270        }
271    
272        private void processRootDocument(HttpServletRequest request, HttpServletResponse response, String pathInfo)
273                throws ServletException, IOException {
274    
275            String sessionId = pathInfo.substring(ROOT_PATH.length());
276    
277            try {
278                Session<VoiceXmlInputTurn, VoiceXmlOutputTurn, VoiceXmlFirstTurn, VoiceXmlLastTurn, VoiceXmlDialogueContext> session = getExistingSession(sessionId);
279                Document rootDocument = mRootDocumentFactory.getDocument(request, session);
280                response.setContentType(VOICE_XML_CONTENT_TYPE);
281                DomUtils.writeToOutputStream(rootDocument, response.getOutputStream());
282            } catch (SessionNotFoundException exception) {
283                throw new ServletException("Unable to find session.", exception);
284            } catch (VoiceXmlDocumentRenderingException exception) {
285                throw new ServletException("Error while rendering root document.", exception);
286            }
287        }
288    
289        private String getETag(byte[] bytes) throws ServletException {
290            try {
291                byte[] digest = MessageDigest.getInstance("MD5").digest(bytes);
292                return StringUtils.bytesToHex(digest);
293            } catch (NoSuchAlgorithmException exception) {
294                throw new ServletException("Could not create message digest.", exception);
295            }
296        }
