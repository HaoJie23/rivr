086    private static final Set<String> RESERVED = new HashSet<String>(Arrays.asList(WORDS));
087
088    /**
089     * Checks if a name is a valid VoiceXML identifier as per VoiceXML 2.0
090     * specification (section 5.1): <blockquote>The variable naming convention
091     * is as in ECMAScript, but names beginning with the underscore character
092     * ("_") and names ending with a dollar sign ("$") are reserved for internal
093     * use. VoiceXML variables, including form item variables, must not contain
094     * ECMAScript reserved words. They must also follow ECMAScript rules for
095     * referential correctness. For example, variable names must be unique and
096     * their declaration must not include a dot - "var x.y" is an illegal
097     * declaration in ECMAScript. Variable names which violate naming
098     * conventions or ECMAScript rules cause an 'error.semantic' event to be
099     * thrown. </blockquote>
100     * 
101     * @param name the name to be tested
102     * @return <code>true</code> if and only if the name is a valid VoiceXML
103     *         identifier.
104     */
105
106    public static boolean isValidIdentifierName(String name) {
107        Assert.notNull(name, "name");
108        return ALLOWED.matcher(name).matches()
109               && !RESERVED.contains(name)
110               && !name.startsWith("_")
111               && !name.startsWith("$");
112    }
