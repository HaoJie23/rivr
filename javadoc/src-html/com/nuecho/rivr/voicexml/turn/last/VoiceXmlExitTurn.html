005    package com.nuecho.rivr.voicexml.turn.last;
006    
007    import static com.nuecho.rivr.voicexml.rendering.voicexml.VoiceXmlDomUtil.*;
008    
009    import javax.json.*;
010    
011    import org.w3c.dom.*;
012    
013    import com.nuecho.rivr.core.util.*;
014    import com.nuecho.rivr.voicexml.rendering.voicexml.*;
015    import com.nuecho.rivr.voicexml.turn.*;
016    import com.nuecho.rivr.voicexml.util.json.*;
017    
018    /**
019     * @author Nu Echo Inc.
020     */
021    public class VoiceXmlExitTurn extends VoiceXmlLastTurn {
022    
023        private static final String VARIABLES_PROPERTY = "variables";
024        private static final String EXPRESSION_PROPERTY = "expression";
025    
026        private VariableDeclarationList mVariables;
027        private String mExpression;
028    
029        public VoiceXmlExitTurn(String name) {
030            super(name);
031        }
032    
033        public VoiceXmlExitTurn(String name, VariableDeclarationList variables) {
034            super(name);
035            Assert.notNull(variables, VARIABLES_PROPERTY);
036            mVariables = variables;
037        }
038    
039        public VoiceXmlExitTurn(String name, String expression) {
040            super(name);
041            mExpression = expression;
042        }
043    
044        @Override
045        protected void addTurnProperties(JsonObjectBuilder builder) {
046            JsonUtils.add(builder, VARIABLES_PROPERTY, mVariables);
047            JsonUtils.add(builder, EXPRESSION_PROPERTY, mExpression);
048        }
049    
050        @Override
051        protected void fillVoiceXmlDocument(Document document, Element formElement, VoiceXmlDialogueContext dialogueContext)
052                throws VoiceXmlDocumentRenderingException {
053            Element blockElement = DomUtils.appendNewElement(formElement, BLOCK_ELEMENT);
054            Element exitElement = document.createElement(EXIT_ELEMENT);
055            if (mVariables != null) {
056                addNamelist(blockElement, exitElement, mVariables);
057            } else if (mExpression != null) {
058                exitElement.setAttribute(EXPR_ATTRIBUTE, mExpression);
059            }
060    
061            blockElement.appendChild(exitElement);
062        }
063    }
