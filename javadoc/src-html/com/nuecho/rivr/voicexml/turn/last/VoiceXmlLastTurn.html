005    package com.nuecho.rivr.voicexml.turn.last;
006    
007    import static com.nuecho.rivr.voicexml.rendering.voicexml.VoiceXmlDomUtil.*;
008    
009    import org.w3c.dom.*;
010    
011    import com.nuecho.rivr.core.channel.*;
012    import com.nuecho.rivr.core.util.*;
013    import com.nuecho.rivr.voicexml.rendering.voicexml.*;
014    import com.nuecho.rivr.voicexml.turn.*;
015    
016    /**
017     * @author Nu Echo Inc.
018     */
019    public abstract class VoiceXmlLastTurn extends VoiceXmlDocumentTurn implements LastTurn {
020    
021        public VoiceXmlLastTurn(String name) {
022            super(name);
023        }
024    
025        @Override
026        protected Document createVoiceXmlDocument(VoiceXmlDialogueContext dialogueContext)
027                throws VoiceXmlDocumentRenderingException {
028            Document document = createDocument(dialogueContext, this);
029            Element formElement = createForm(document);
030            fillVoiceXmlDocument(document, formElement, dialogueContext);
031            addEventHandlers(document.getDocumentElement());
032            addFatalErrorHandlerForm(dialogueContext, document, this);
033            return document;
034        }
035    
036        private void addEventHandlers(Element vxmlElement) {
037            Element catchElement = DomUtils.appendNewElement(vxmlElement, CATCH_ELEMENT);
038            catchElement.setAttribute(EVENT_ATTRIBUTE, ERROR_EVENT_NAME);
039            createGotoFatalHandler(catchElement);
040        }
041    
042        protected abstract void fillVoiceXmlDocument(Document document,
043                                                     Element formElement,
044                                                     VoiceXmlDialogueContext dialogueContext)
045                throws VoiceXmlDocumentRenderingException;
046    
047    }
