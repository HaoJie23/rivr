006    
007    import static com.nuecho.rivr.voicexml.rendering.voicexml.VoiceXmlDomUtil.*;
008    
009    import javax.json.*;
010    
011    import org.w3c.dom.*;
012    
013    import com.nuecho.rivr.core.util.*;
014    import com.nuecho.rivr.voicexml.rendering.voicexml.*;
015    import com.nuecho.rivr.voicexml.turn.*;
016    import com.nuecho.rivr.voicexml.util.json.*;
017    
018    /**
019     * @author Nu Echo Inc.
020     */
021    public class VoiceXmlDisconnectTurn extends VoiceXmlLastTurn {
022    
023        private static final String VARIABLES_PROPERTY = "variables";
024    
025        private VariableDeclarationList mVariables;
026    
027        public VoiceXmlDisconnectTurn(String name) {
028            super(name);
029        }
030    
031        public VoiceXmlDisconnectTurn(String name, VariableDeclarationList variables) {
032            super(name);
033            Assert.notNull(variables, VARIABLES_PROPERTY);
034            mVariables = variables;
035        }
036    
037        @Override
038        protected void addTurnProperties(JsonObjectBuilder builder) {
039            JsonUtils.add(builder, VARIABLES_PROPERTY, mVariables);
040        }
041    
042        @Override
043        protected void fillVoiceXmlDocument(Document document, Element formElement, VoiceXmlDialogueContext dialogueContext)
044                throws VoiceXmlDocumentRenderingException {
045            Element blockElement = DomUtils.appendNewElement(formElement, BLOCK_ELEMENT);
046            Element disconnectElement = document.createElement(DISCONNECT_ELEMENT);
047    
048            if (mVariables != null) {
049                addNamelist(blockElement, disconnectElement, mVariables);
050            }
051    
052            blockElement.appendChild(disconnectElement);
053        }
054    }
