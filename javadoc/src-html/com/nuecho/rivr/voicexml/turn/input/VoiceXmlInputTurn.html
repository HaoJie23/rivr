004    package com.nuecho.rivr.voicexml.turn.input;
005    
006    import static java.util.Collections.*;
007    
008    import java.util.*;
009    
010    import javax.json.*;
011    
012    import com.nuecho.rivr.core.channel.*;
013    import com.nuecho.rivr.core.util.*;
014    import com.nuecho.rivr.voicexml.util.json.*;
015    
016    public final class VoiceXmlInputTurn implements InputTurn, JsonSerializable {
017    
018        private static final String TRANSFER_RESULT_PROPERTY = "transferResult";
019        private static final String RECORDING_INFO_PROPERTY = "recordingInfo";
020        private static final String RECOGNITION_INFO_PROPERTY = "recognitionInfo";
021        private static final String EVENTS_PROPERTY = "events";
022        private static final String VALUE_PROPERTY = "value";
023    
024        private List<VoiceXmlEvent> mEvents = emptyList();
025        private JsonValue mJsonValue;
026        private RecognitionInfo mRecognitionInfo;
027        private RecordingInfo mRecordingInfo;
028        private TransferStatusInfo mTransferResult;
029    
030        public List<VoiceXmlEvent> getEvents() {
031            return Collections.unmodifiableList(mEvents);
032        }
033    
034        public JsonValue getJsonValue() {
035            return mJsonValue;
036        }
037    
038        public RecognitionInfo getRecognitionInfo() {
039            return mRecognitionInfo;
040        }
041    
042        public RecordingInfo getRecordingInfo() {
043            return mRecordingInfo;
044        }
045    
046        public TransferStatusInfo getTransferResult() {
047            return mTransferResult;
048        }
049    
050        public void setEvents(List<VoiceXmlEvent> events) {
051            Assert.notNull(events, "events");
052            mEvents = new ArrayList<VoiceXmlEvent>(events);
053        }
054    
055        public void setJsonValue(JsonValue jsonValue) {
056            mJsonValue = jsonValue;
057        }
058    
059        public void setRecognitionInfo(RecognitionInfo recognitionInfo) {
060            mRecognitionInfo = recognitionInfo;
061        }
062    
063        public void setRecordingInfo(RecordingInfo recordingInfo) {
064            mRecordingInfo = recordingInfo;
065        }
066    
067        public void setTransferResult(TransferStatusInfo transferResult) {
068            mTransferResult = transferResult;
069        }
070    
071        @Override
072        public String toString() {
073            return asJson().toString();
074        }
075    
076        @Override
077        public JsonValue asJson() {
078            JsonObjectBuilder builder = JsonUtils.createObjectBuilder();
079            JsonUtils.add(builder, VALUE_PROPERTY, mJsonValue);
080            JsonUtils.add(builder, EVENTS_PROPERTY, JsonUtils.toJson(mEvents));
081            JsonUtils.add(builder, RECOGNITION_INFO_PROPERTY, mRecognitionInfo);
082            JsonUtils.add(builder, RECORDING_INFO_PROPERTY, mRecordingInfo);
083            JsonUtils.add(builder, TRANSFER_RESULT_PROPERTY, mTransferResult);
084            return builder.build();
085        }
086    }
