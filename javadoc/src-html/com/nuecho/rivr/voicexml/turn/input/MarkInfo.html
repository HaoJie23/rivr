001    package com.nuecho.rivr.voicexml.turn.input;
002    
003    import javax.json.*;
004    
005    import com.nuecho.rivr.core.util.*;
006    import com.nuecho.rivr.voicexml.util.json.*;
007    
008    public final class MarkInfo implements JsonSerializable {
009    
010        private static final String TIME_PROPERTY = "time";
011        private static final String NAME_PROPERTY = "name";
012        private final String mName;
013        private final long mTime;
014    
015        public MarkInfo(String name, long time) {
016            Assert.notEmpty(name, NAME_PROPERTY);
017            Assert.notNegative(time, TIME_PROPERTY);
018            mName = name;
019            mTime = time;
020        }
021    
022        public String getName() {
023            return mName;
024        }
025    
026        public long getTime() {
027            return mTime;
028        }
029    
030        @Override
031        public JsonValue asJson() {
032            JsonObjectBuilder builder = JsonUtils.createObjectBuilder();
033            JsonUtils.add(builder, NAME_PROPERTY, mName);
034            builder.add(TIME_PROPERTY, mTime);
035            return builder.build();
036        }
037    }
