006    import javax.json.*;
007    
008    import com.nuecho.rivr.voicexml.util.json.*;
009    
010    /**
011     * @author NuEcho Inc.
012     */
013    public class TransferStatus implements JsonSerializable {
014    
015        public static final String NO_ANSWER = "noanswer";
016        public static final String NEAR_END_DISCONNECT = "near_end_disconnect";
017        public static final String FAR_END_DISCONNECT = "far_end_disconnect";
018        public static final String NETWORK_DISONNECT = "network_disconnect";
019        public static final String MAXTIME_DISCONNECT = "maxtime_disconnect";
020        public static final String BUSY = "busy";
021        public static final String NETWORK_BUSY = "network_busy";
022        public static final String UNKNOWN = "unknown";
023    
024        private final String mStatusCode;
025    
026        public TransferStatus(String statusCode) {
027            mStatusCode = statusCode;
028        }
029    
030        public String getStatusCode() {
031            return mStatusCode;
032        }
033    
034        @Override
035        public JsonValue asJson() {
036            return JsonUtils.wrap(mStatusCode);
037        }
038    
039    }
