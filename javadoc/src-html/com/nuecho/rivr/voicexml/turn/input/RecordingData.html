001    package com.nuecho.rivr.voicexml.turn.input;
002    
003    import javax.json.*;
004    
005    import com.nuecho.rivr.voicexml.util.json.*;
006    
007    public final class RecordingData implements JsonSerializable {
008    
009        private static final String SIZE_PROPERTY = "size";
010        private static final String TYPE_PROPERTY = "type";
011        private static final String FILENAME_PROPERTY = "filename";
012    
013        private final byte[] mAudioPayLoad;
014        private final String mType;
015        private final String mFilename;
016    
017        public RecordingData(byte[] audioPayLoad, String type, String fileName) {
018            mAudioPayLoad = audioPayLoad;
019            mType = type;
020            mFilename = fileName;
021        }
022    
023        public byte[] getAudioPayLoad() {
024            return mAudioPayLoad;
025        }
026    
027        public String getType() {
028            return mType;
029        }
030    
031        public String getFilename() {
032            return mFilename;
033        }
034    
035        @Override
036        public String toString() {
037            return asJson().toString();
038        }
039    
040        @Override
041        public JsonValue asJson() {
042            JsonObjectBuilder builder = JsonUtils.createObjectBuilder();
043            JsonUtils.add(builder, FILENAME_PROPERTY, mFilename);
044            JsonUtils.add(builder, TYPE_PROPERTY, mType);
045            if (mAudioPayLoad != null) {
046                builder.add(SIZE_PROPERTY, (long) mAudioPayLoad.length);
047            }
048            //payload is not serialized
049            return builder.build();
050        }
051    }
