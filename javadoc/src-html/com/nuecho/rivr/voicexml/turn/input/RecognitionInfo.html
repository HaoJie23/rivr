004    package com.nuecho.rivr.voicexml.turn.input;
005    
006    import javax.json.*;
007    
008    import com.nuecho.rivr.core.util.*;
009    import com.nuecho.rivr.voicexml.util.json.*;
010    
011    /**
012     * @author Nu Echo Inc.
013     */
014    public final class RecognitionInfo implements JsonSerializable {
015    
016        private static final String MARK_INFO_PROPERTY = "markInfo";
017        private static final String RECOGNITION_RESULT_PROPERTYY = "recognitionResult";
018        private final JsonArray mRecognitionResult;
019        private final MarkInfo mMarkInfo;
020    
021        public RecognitionInfo(JsonArray recognitionResult, MarkInfo markInfo) {
022            Assert.notNull(recognitionResult, RECOGNITION_RESULT_PROPERTYY);
023            mRecognitionResult = recognitionResult;
024            mMarkInfo = markInfo;
025        }
026    
027        public JsonArray getRecognitionResult() {
028            return mRecognitionResult;
029        }
030    
031        public MarkInfo getMarkInfo() {
032            return mMarkInfo;
033        }
034    
035        @Override
036        public String toString() {
037            return asJson().toString();
038        }
039    
040        @Override
041        public JsonValue asJson() {
042            JsonObjectBuilder builder = JsonUtils.createObjectBuilder();
043            JsonUtils.add(builder, RECOGNITION_RESULT_PROPERTYY, mRecognitionResult);
044            JsonUtils.add(builder, MARK_INFO_PROPERTY, mMarkInfo);
045            return builder.build();
046        }
047    }
