005    package com.nuecho.rivr.voicexml.turn;
006    
007    import java.util.*;
008    
009    import javax.json.*;
010    
011    import com.nuecho.rivr.core.util.*;
012    import com.nuecho.rivr.voicexml.rendering.voicexml.*;
013    import com.nuecho.rivr.voicexml.util.json.*;
014    
015    /**
016     * @author Nu Echo Inc.
017     */
018    public final class VariableDeclarationList implements Iterable<VariableDeclaration>, JsonSerializable {
019    
020        private static final String VARIABLE_NAME_PROPERTY = "name";
021        private static final String INITIAL_VALUE_PROPERTY = "initialValue";
022    
023        private final LinkedHashMap<String, VariableDeclaration> mVariables = new LinkedHashMap<String, VariableDeclaration>();
024    
025        public void addVariable(VariableDeclaration variableDeclaration) {
026            Assert.notNull(variableDeclaration, "variableDeclaration");
027            String name = variableDeclaration.getName();
028            if (mVariables.containsKey(name)) throw new IllegalArgumentException("Variable " + name + " already present.");
029    
030            mVariables.put(name, variableDeclaration);
031        }
032    
033        public void removeVariable(VariableDeclaration variableDeclaration) {
034            Assert.notNull(variableDeclaration, "variableDeclaration");
035            mVariables.remove(variableDeclaration.getName());
036        }
037    
038        public boolean has(String variableName) {
039            return mVariables.containsKey(variableName);
040        }
041    
042        public VariableDeclaration get(String variableName) {
043            return mVariables.get(variableName);
044        }
045    
046        public boolean isEmpty() {
047            return mVariables.isEmpty();
048        }
049    
050        @Override
051        public int hashCode() {
052            final int prime = 31;
053            int result = 1;
054            result = prime * result + ((mVariables == null) ? 0 : mVariables.hashCode());
055            return result;
056        }
057    
058        @Override
059        public boolean equals(Object obj) {
060            if (this == obj) return true;
061            if (obj == null) return false;
062            if (getClass() != obj.getClass()) return false;
063            VariableDeclarationList other = (VariableDeclarationList) obj;
064            if (mVariables == null) {
065                if (other.mVariables != null) return false;
066            } else if (!mVariables.equals(other.mVariables)) return false;
067            return true;
068        }
069    
070        public static VariableDeclarationList create(JsonObject jsonObject) {
071            VariableDeclarationList list = new VariableDeclarationList();
072            for (String propertyName : jsonObject.keySet()) {
073                JsonValue jsonValue = jsonObject.get(propertyName);
074    
075                String ecmaScriptLiteral;
076                if (jsonValue instanceof JsonString) {
077                    JsonString jsonString = (JsonString) jsonValue;
078                    ecmaScriptLiteral = VoiceXmlDomUtil.createEcmaScriptStringLiteral(jsonString.getString());
079                } else {
080                    ecmaScriptLiteral = jsonValue.toString();
081                }
082    
083                list.addVariable(new VariableDeclaration(propertyName, ecmaScriptLiteral));
084            }
085            return list;
086        }
087    
088        @Override
089        public String toString() {
090            return asJson().toString();
091        }
092    
093        @Override
094        public JsonValue asJson() {
095            JsonArrayBuilder builder = JsonUtils.createArrayBuilder();
096            for (VariableDeclaration variableDeclaration : this) {
097                JsonObjectBuilder variableBuilder = JsonUtils.createObjectBuilder();
098                JsonUtils.add(variableBuilder, VARIABLE_NAME_PROPERTY, variableDeclaration.getName());
099                if (variableDeclaration.getInitialValueExpression() != null) {
100                    JsonUtils.add(variableBuilder, INITIAL_VALUE_PROPERTY, variableDeclaration.getInitialValueExpression());
101                }
102    
103                builder.add(variableBuilder);
104            }
105            return builder.build();
106        }
107    
108        @Override
109        public Iterator<VariableDeclaration> iterator() {
110            return Collections.unmodifiableCollection(mVariables.values()).iterator();
111        }
112    }
