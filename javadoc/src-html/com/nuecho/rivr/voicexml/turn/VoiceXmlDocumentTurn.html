005    package com.nuecho.rivr.voicexml.turn;
006    
007    import java.util.*;
008    
009    import javax.json.*;
010    
011    import org.w3c.dom.*;
012    
013    import com.nuecho.rivr.core.util.*;
014    import com.nuecho.rivr.voicexml.rendering.voicexml.*;
015    import com.nuecho.rivr.voicexml.util.json.*;
016    
017    /**
018     * @author Nu Echo Inc.
019     */
020    public abstract class VoiceXmlDocumentTurn implements JsonSerializable, NamedTurn {
021    
022        private static final String DATA_PROPERTY = "data";
023        private static final String NAME_PROPERTY = "name";
024    
025        private final String mName;
026    
027        private final List<VoiceXmlDocumentAdapter> mAdapters = new ArrayList<VoiceXmlDocumentAdapter>();
028    
029        public VoiceXmlDocumentTurn(String name) {
030            Assert.notEmpty(name, "name");
031            mName = name;
032        }
033    
034        protected abstract Document createVoiceXmlDocument(VoiceXmlDialogueContext dialogueContext)
035                throws VoiceXmlDocumentRenderingException;
036    
037        /**
038         * @param builder
039         */
040        protected void addTopLevelProperties(JsonObjectBuilder builder) {}
041    
042        protected abstract void addTurnProperties(JsonObjectBuilder builder);
043    
044        public final void addAdapter(VoiceXmlDocumentAdapter adapter) {
045            mAdapters.add(adapter);
046        }
047    
048        @Override
049        public final String getName() {
050            return mName;
051        }
052    
053        public final Document getVoiceXmlDocument(VoiceXmlDialogueContext dialogueContext)
054                throws VoiceXmlDocumentRenderingException {
055            Document document = createVoiceXmlDocument(dialogueContext);
056            for (VoiceXmlDocumentAdapter adapter : mAdapters) {
057                adapter.adaptVoiceXmlDocument(document);
058            }
059            return document;
060        }
061    
062        @Override
063        public final String toString() {
064            return asJson().toString();
065        }
066    
067        @Override
068        public final JsonValue asJson() {
069            JsonObjectBuilder builder = JsonUtils.createObjectBuilder();
070            JsonObjectBuilder dataBuilder = JsonUtils.createObjectBuilder();
071            addTurnProperties(dataBuilder);
072    
073            JsonUtils.add(builder, NAME_PROPERTY, getName());
074            JsonUtils.add(builder, DATA_PROPERTY, dataBuilder.build());
075            addTopLevelProperties(builder);
076            return builder.build();
077        }
078    }
