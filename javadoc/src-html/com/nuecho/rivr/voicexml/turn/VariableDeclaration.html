005    package com.nuecho.rivr.voicexml.turn;
006    
007    import com.nuecho.rivr.core.util.*;
008    import com.nuecho.rivr.voicexml.util.*;
009    
010    /**
011     * @author Nu Echo Inc.
012     */
013    public final class VariableDeclaration {
014    
015        private final String mName;
016        private final String mInitialValueExpression;
017    
018        public VariableDeclaration(String name, String initialValueExpression) {
019            Assert.notNull(name, "name");
020            Assert.ensure(VoiceXmlUtils.isValidIdentifierName(name), "Invalid ECMAScript identifier name: '" + name + "'");
021            mName = name;
022            mInitialValueExpression = initialValueExpression;
023        }
024    
025        public VariableDeclaration(String name) {
026            this(name, null);
027        }
028    
029        public String getName() {
030            return mName;
031        }
032    
033        public String getInitialValueExpression() {
034            return mInitialValueExpression;
035        }
036    
037        @Override
038        public String toString() {
039            ToStringBuilder builder = new ToStringBuilder(this);
040            builder.appendItem("mName", mName);
041            builder.appendItem("mInitialValueExpression", mInitialValueExpression);
042            return builder.getString();
043        }
044    
045    }
