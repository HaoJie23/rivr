005    package com.nuecho.rivr.voicexml.turn.first;
006    
007    import java.util.*;
008    
009    import com.nuecho.rivr.core.channel.*;
010    
011    /**
012     * @author Nu Echo Inc.
013     */
014    public class VoiceXmlFirstTurn implements FirstTurn {
015    
016        private final Map<String, String> mParameters;
017    
018        public VoiceXmlFirstTurn() {
019            mParameters = Collections.emptyMap();
020        }
021    
022        public VoiceXmlFirstTurn(Map<String, String> parameters) {
023            mParameters = new HashMap<String, String>(parameters);
024        }
025    
026        public Map<String, String> getParameters() {
027            return Collections.unmodifiableMap(mParameters);
028        }
029    
030        public boolean hasParameter(Object key) {
031            return mParameters.containsKey(key);
032        }
033    
034        public String getParameters(Object key) {
035            return mParameters.get(key);
036        }
037    
038        public Set<String> keySet() {
039            return mParameters.keySet();
040        }
041    }
