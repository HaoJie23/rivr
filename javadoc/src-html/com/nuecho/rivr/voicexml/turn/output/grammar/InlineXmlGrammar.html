004    package com.nuecho.rivr.voicexml.turn.output.grammar;
005    
006    import javax.json.*;
007    
008    import org.w3c.dom.*;
009    
010    import com.nuecho.rivr.core.util.*;
011    import com.nuecho.rivr.voicexml.util.json.*;
012    
013    /**
014     * An <code>InlineXmlGrammar</code> represents an XML source grammar inlined in
015     * the VoiceXML document.
016     * 
017     * @author Nu Echo Inc.
018     */
019    public final class InlineXmlGrammar extends GrammarItem {
020        private static final String SOURCE_PROPERTY = "source";
021        private static final String INLINE_XML_ELEMENT_TYPE = "inlineXml";
022    
023        private final Document mDocument;
024    
025        /**
026         * @param document The XML source of the grammar. Not null.
027         */
028        public InlineXmlGrammar(Document document) {
029            Assert.notNull(document, "document");
030            mDocument = document;
031        }
032    
033        public Document getDocument() {
034            return mDocument;
035        }
036    
037        @Override
038        public String getElementType() {
039            return INLINE_XML_ELEMENT_TYPE;
040        }
041    
042        @Override
043        protected void addJsonProperties(JsonObjectBuilder builder) {
044            JsonUtils.addXmlNodeProperty(builder, SOURCE_PROPERTY, "grammar document", mDocument);
045        }
046    
047        @Override
048        public int hashCode() {
049            final int prime = 31;
050            int result = super.hashCode();
051            result = prime * result + (mDocument == null ? 0 : mDocument.hashCode());
052            return result;
053        }
054    
055        @Override
056        public boolean equals(Object obj) {
057            if (this == obj) return true;
058            if (!super.equals(obj)) return false;
059            if (getClass() != obj.getClass()) return false;
060            InlineXmlGrammar other = (InlineXmlGrammar) obj;
061            if (mDocument == null) {
062                if (other.mDocument != null) return false;
063            } else if (!mDocument.equals(other.mDocument)) return false;
064            return true;
065        }
066    }
