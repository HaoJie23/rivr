006    import static com.nuecho.rivr.voicexml.rendering.voicexml.VoiceXmlDomUtil.*;

010    import org.w3c.dom.*;
011    
012    import com.nuecho.rivr.core.util.*;
013    import com.nuecho.rivr.voicexml.rendering.voicexml.*;
014    import com.nuecho.rivr.voicexml.turn.output.*;
015    import com.nuecho.rivr.voicexml.util.json.*;

017    /**
018     * This abstract class is the superclass of all classes representing a transfer
019     * to another entity.
020     * 
021     * @author Nu Echo Inc.
022     * @see BlindTransferTurn
023     * @see BridgeTransferTurn
024     * @see ConsultationTransferTurn
025     * @see <a
026     *      href="http://www.w3.org/TR/voicexml20/#dml2.3.6">http://www.w3.org/TR/voicexml20/#dml2.3.6</a>
027     * @see <a
028     *      href="http://www.w3.org/TR/voicexml21/#sec-transfer">http://www.w3.org/TR/voicexml21/#sec-transfer</a>
029     */
030    public abstract class TransferTurn extends VoiceXmlOutputTurn {
031        private static final String TRANSFER_TURN_TYPE = "transfer";
032    
033        private static final String TRANSFER_TYPE_PROPERTY = "transferType";
034        private static final String APPLICATION_TO_APPLICATION_INFORMATION_PROPERTY = "applicationToApplicationInformation";
035        private static final String DESTINATION_PROPERTY = "destination";
036    
037        private final String mDestination;
038        private String mApplicationToApplicationInformation;
039    
040        /**
041         * @param name The name of this turn. Not empty.
042         * @param destination The URI of the destination (telephone, IP telephony
043         *            address). Not empty.
044         */
045        public TransferTurn(String name, String destination) {
046            super(name);
047            Assert.notEmpty(destination, "destination");
048            mDestination = destination;
049        }
050    
051        /**
052         * @param applicationToApplicationInformation A string containing data sent
053         *            to an application on the far-end, available in the session
054         *            variable session.connection.aai.
055         */
056        public final void setApplicationToApplicationInformation(String applicationToApplicationInformation) {
057            mApplicationToApplicationInformation = applicationToApplicationInformation;
058        }
059    
060        public final String getDestination() {
061            return mDestination;
062        }
063    
064        public final String getApplicationToApplicationInformation() {
065            return mApplicationToApplicationInformation;
066        }
067    
068        protected abstract String getTransferType();
069    
070        /**
071         * @param transferElement The transfer element to customize.
072         * @throws VoiceXmlDocumentRenderingException
073         */
074        protected void customizeTransferElement(Element transferElement) throws VoiceXmlDocumentRenderingException {}
075    
076        @Override
077        protected final String getOuputTurnType() {
078            return TRANSFER_TURN_TYPE;
079        }
080    
081        @Override
082        protected void addTurnProperties(JsonObjectBuilder builder) {
083            JsonUtils.add(builder, DESTINATION_PROPERTY, mDestination);
084            JsonUtils.add(builder, APPLICATION_TO_APPLICATION_INFORMATION_PROPERTY, mApplicationToApplicationInformation);
085            JsonUtils.add(builder, TRANSFER_TYPE_PROPERTY, getTransferType());
086        }
087    
088        @Override
089        protected void fillVoiceXmlDocument(Document document, Element formElement, VoiceXmlDialogueContext dialogueContext)
090                throws VoiceXmlDocumentRenderingException {
091    
092            Element transferElement = DomUtils.appendNewElement(formElement, TRANSFER_ELEMENT);
093    
094            transferElement.setAttribute(TYPE_ATTRIBUTE, getTransferType());
095            transferElement.setAttribute(NAME_ATTRIBUTE, TRANSFER_FORM_ITEM_NAME);
096            transferElement.setAttribute(DEST_ATTRIBUTE, mDestination);
097    
098            if (mApplicationToApplicationInformation != null) {
099                transferElement.setAttribute(AAI_ATTRIBUTE, mApplicationToApplicationInformation);
100            }
101    
102            customizeTransferElement(transferElement);
103    
104            Element filledElement = DomUtils.appendNewElement(transferElement, FILLED_ELEMENT);
105            String script = RIVR_SCOPE_OBJECT
106                            + ".addTransferResult("
107                            + TRANSFER_FORM_ITEM_NAME
108                            + ", "
109                            + TRANSFER_FORM_ITEM_NAME
110                            + "$);";
111            createScript(filledElement, script);
112            createGotoSubmit(filledElement);
113        }
114    }
