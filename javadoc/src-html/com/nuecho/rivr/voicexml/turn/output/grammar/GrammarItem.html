005    package com.nuecho.rivr.voicexml.turn.output.grammar;
006    
007    import javax.json.*;
008    
009    import com.nuecho.rivr.voicexml.util.json.*;
010    
011    /**
012     * This abstract class is the superclass of all classes representing a grammar
013     * element in the dialogue. It is not intended to be subclassed.
014     * 
015     * @author Nu Echo Inc.
016     * @see GrammarReference
017     * @see InlineStringGrammar
018     * @see InlineXmlGrammar
019     * @see <a href="http://www.w3.org/TR/voicexml20/#dml3.1">http://www.w3.org/TR/voicexml20/#dml3.1</a>
020     */
021    public abstract class GrammarItem implements JsonSerializable {
022        public static final String TYPE_PROPERTY = "type";
023    
024        private Double mWeight;
025        private String mMediaType;
026    
027        /**
028         * @param weight The relative weight of this grammar. Null reverts to
029         *            VoiceXML default value.
030         * @see <a href="http://www.w3.org/TR/voicexml20/#dml3.1.1.3">http://www.w3.org/TR/voicexml20/#dml3.1.1.3</a>
031         */
032        public final void setWeight(Double weight) {
033            mWeight = weight;
034        }
035    
036        /**
037         * @param mediaType The preferred media type for this grammar. Null reverts
038         *            to VoiceXML default value.
039         * @see <a href="http://www.w3.org/TR/voicexml20/#dml3.1.1.4">http://www.w3.org/TR/voicexml20/#dml3.1.1.4</a>
040         */
041        public final void setMediaType(String mediaType) {
042            mMediaType = mediaType;
043        }
044    
045        public final Double getWeight() {
046            return mWeight;
047        }
048    
049        public final String getMediaType() {
050            return mMediaType;
051        }
052    
053        public abstract String getElementType();
054    
055        protected abstract void addJsonProperties(JsonObjectBuilder builder);
056    
057        @Override
058        public final String toString() {
059            return asJson().toString();
060        }
061    
062        @Override
063        public final JsonValue asJson() {
064            JsonObjectBuilder builder = JsonUtils.createObjectBuilder();
065            JsonUtils.add(builder, TYPE_PROPERTY, getElementType());
066            addJsonProperties(builder);
067            return builder.build();
068        }
069    
070        @Override
071        public int hashCode() {
072            final int prime = 31;
073            int result = 1;
074            result = prime * result + (mMediaType == null ? 0 : mMediaType.hashCode());
075            result = prime * result + (mWeight == null ? 0 : mWeight.hashCode());
076            return result;
077        }
078    
079        @Override
080        public boolean equals(Object obj) {
081            if (this == obj) return true;
082            if (obj == null) return false;
083            if (getClass() != obj.getClass()) return false;
084            GrammarItem other = (GrammarItem) obj;
085            if (mMediaType == null) {
086                if (other.mMediaType != null) return false;
087            } else if (!mMediaType.equals(other.mMediaType)) return false;
088            if (mWeight == null) {
089                if (other.mWeight != null) return false;
090            } else if (!mWeight.equals(other.mWeight)) return false;
091            return true;
092        }
093    }
