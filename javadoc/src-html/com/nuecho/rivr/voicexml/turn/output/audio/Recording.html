004    package com.nuecho.rivr.voicexml.turn.output.audio;
005    
006    import javax.json.*;
007    
008    import com.nuecho.rivr.core.util.*;
009    import com.nuecho.rivr.voicexml.turn.output.fetch.*;
010    import com.nuecho.rivr.voicexml.util.json.*;
011    
012    /**
013     * A <code>Recording</code> represents an audio file with an optional synthesis
014     * text backup.
015     * 
016     * @author Nu Echo Inc.
017     * @see SynthesisText
018     * @see ResourceFetchConfiguration
019     */
020    public final class Recording extends AudioItem {
021        public static final String RECORDING_ELEMENT_TYPE = "recording";
022        private static final String RESOURCE_FETCH_CONFIGURATION_PROPERTY = "resourceFetchConfiguration";
023        private static final String PATH_PROPERTY = "path";
024    
025        private final String mPath;
026        private final SynthesisText mSynthesisText;
027        private ResourceFetchConfiguration mResourceFetchConfiguration;
028    
029        /**
030         * @param path The URI of the audio prompt. Not empty.
031         */
032        public Recording(String path) {
033            this(path, null);
034        }
035    
036        /**
037         * @param path The URI of the audio prompt. Not empty.
038         * @param synthesisText The synthesis text to play if the audio prompt
039         *            cannot be found.
040         */
041        public Recording(String path, SynthesisText synthesisText) {
042            Assert.notEmpty(path, "path");
043            mPath = path;
044            mSynthesisText = synthesisText;
045        }
046    
047        /**
048         * @param resourceFetchConfiguration The resource fetch configuration. Null
049         *            reverts to VoiceXML default value.
050         */
051        public void setResourceFetchConfiguration(ResourceFetchConfiguration resourceFetchConfiguration) {
052            mResourceFetchConfiguration = resourceFetchConfiguration;
053        }
054    
055        @Override
056        public String getElementType() {
057            return RECORDING_ELEMENT_TYPE;
058        }
059    
060        public String getPath() {
061            return mPath;
062        }
063    
064        public SynthesisText getSynthesisText() {
065            return mSynthesisText;
066        }
067    
068        public ResourceFetchConfiguration getResourceFetchConfiguration() {
069            return mResourceFetchConfiguration;
070        }
071    
072        @Override
073        protected void addJsonProperties(JsonObjectBuilder builder) {
074            JsonUtils.add(builder, PATH_PROPERTY, mPath);
075            JsonUtils.add(builder, RESOURCE_FETCH_CONFIGURATION_PROPERTY, mResourceFetchConfiguration);
076        }
077    
078        @Override
079        public int hashCode() {
080            final int prime = 31;
081            int result = 1;
082            result = prime * result + (mPath == null ? 0 : mPath.hashCode());
083            result = prime * result + (mResourceFetchConfiguration == null ? 0 : mResourceFetchConfiguration.hashCode());
084            result = prime * result + (mSynthesisText == null ? 0 : mSynthesisText.hashCode());
085            return result;
086        }
087    
088        @Override
089        public boolean equals(Object obj) {
090            if (this == obj) return true;
091            if (obj == null) return false;
092            if (getClass() != obj.getClass()) return false;
093            Recording other = (Recording) obj;
094            if (mPath == null) {
095                if (other.mPath != null) return false;
096            } else if (!mPath.equals(other.mPath)) return false;
097            if (mResourceFetchConfiguration == null) {
098                if (other.mResourceFetchConfiguration != null) return false;
099            } else if (!mResourceFetchConfiguration.equals(other.mResourceFetchConfiguration)) return false;
100            if (mSynthesisText == null) {
101                if (other.mSynthesisText != null) return false;
102            } else if (!mSynthesisText.equals(other.mSynthesisText)) return false;
103            return true;
104        }
105    }
