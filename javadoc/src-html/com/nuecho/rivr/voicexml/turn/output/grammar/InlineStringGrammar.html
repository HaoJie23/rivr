004    package com.nuecho.rivr.voicexml.turn.output.grammar;
005    
006    import javax.json.*;
007    
008    import com.nuecho.rivr.core.util.*;
009    import com.nuecho.rivr.voicexml.util.json.*;
010    
011    /**
012     * An <code>InlineStringGrammar</code> represents a text source grammar inlined
013     * in the VoiceXML document.
014     * 
015     * @author Nu Echo Inc.
016     * @see <a href="http://www.w3.org/TR/voicexml20/#dml3.1.1.4">http://www.w3.org/TR/voicexml20/#dml3.1.1.4</a>
017     */
018    public final class InlineStringGrammar extends GrammarItem {
019        private static final String INLINE_STRING_ELEMENT_TYPE = "inlineString";
020        private static final String VERSION_PROPERTY = "version";
021        private static final String TAG_FORMAT_PROPERTY = "tagFormat";
022        private static final String SOURCE_PROPERTY = "source";
023        private static final String ROOT_PROPERTY = "root";
024        private static final String LANGUAGE_PROPERTY = "language";
025        private static final String BASE_PROPERTY = "base";
026    
027        private final String mSource;
028        private String mVersion;
029        private String mLanguage;
030        private String mRoot;
031        private String mTagFormat;
032        private String mBase;
033    
034        /**
035         * @param source The text source of the grammar. Not null.
036         */
037        public InlineStringGrammar(String source) {
038            Assert.notNull(source, "source");
039            mSource = source;
040        }
041    
042        /**
043         * @param version The version of the grammar. Null reverts to VoiceXML
044         *            default value.
045         */
046        public void setVersion(String version) {
047            mVersion = version;
048        }
049    
050        /**
051         * @param language The language identifier of the grammar. For example,
052         *            "fr-CA" for Canadian French. Null reverts to VoiceXML default
053         *            value.
054         */
055        public void setLanguage(String language) {
056            mLanguage = language;
057        }
058    
059        /**
060         * @param root The name of the rule that will act as the root rule. Null
061         *            reverts to VoiceXML default value.
062         */
063        public void setRoot(String root) {
064            mRoot = root;
065        }
066    
067        /**
068         * @param tagFormat The tag content format for all tags within the grammar.
069         *            Null reverts to VoiceXML default value.
070         */
071        public void setTagFormat(String tagFormat) {
072            mTagFormat = tagFormat;
073        }
074    
075        /**
076         * @param base The base URI from which relative URIs in the grammar are
077         *            resolved. Null reverts to VoiceXML default value.
078         */
079        public void setBase(String base) {
080            mBase = base;
081        }
082    
083        @Override
084        public String getElementType() {
085            return INLINE_STRING_ELEMENT_TYPE;
086        }
087    
088        public String getSource() {
089            return mSource;
090        }
091    
092        public String getVersion() {
093            return mVersion;
094        }
095    
096        public String getLanguage() {
097            return mLanguage;
098        }
099    
100        public String getRoot() {
101            return mRoot;
102        }
103    
104        public String getTagFormat() {
105            return mTagFormat;
106        }
107    
108        public String getBase() {
109            return mBase;
110        }
111    
112        @Override
113        protected void addJsonProperties(JsonObjectBuilder builder) {
114            JsonUtils.add(builder, BASE_PROPERTY, mBase);
115            JsonUtils.add(builder, LANGUAGE_PROPERTY, mLanguage);
116            JsonUtils.add(builder, ROOT_PROPERTY, mRoot);
117            JsonUtils.add(builder, SOURCE_PROPERTY, mSource);
118            JsonUtils.add(builder, TAG_FORMAT_PROPERTY, mTagFormat);
119            JsonUtils.add(builder, VERSION_PROPERTY, mVersion);
120        }
121    
122        @Override
123        public int hashCode() {
124            final int prime = 31;
125            int result = super.hashCode();
126            result = prime * result + (mBase == null ? 0 : mBase.hashCode());
127            result = prime * result + (mLanguage == null ? 0 : mLanguage.hashCode());
128            result = prime * result + (mRoot == null ? 0 : mRoot.hashCode());
129            result = prime * result + (mSource == null ? 0 : mSource.hashCode());
130            result = prime * result + (mTagFormat == null ? 0 : mTagFormat.hashCode());
131            result = prime * result + (mVersion == null ? 0 : mVersion.hashCode());
132            return result;
133        }
134    
135        @Override
136        public boolean equals(Object obj) {
137            if (this == obj) return true;
138            if (!super.equals(obj)) return false;
139            if (getClass() != obj.getClass()) return false;
140            InlineStringGrammar other = (InlineStringGrammar) obj;
141            if (mBase == null) {
142                if (other.mBase != null) return false;
143            } else if (!mBase.equals(other.mBase)) return false;
144            if (mLanguage == null) {
145                if (other.mLanguage != null) return false;
146            } else if (!mLanguage.equals(other.mLanguage)) return false;
147            if (mRoot == null) {
148                if (other.mRoot != null) return false;
149            } else if (!mRoot.equals(other.mRoot)) return false;
150            if (mSource == null) {
151                if (other.mSource != null) return false;
152            } else if (!mSource.equals(other.mSource)) return false;
153            if (mTagFormat == null) {
154                if (other.mTagFormat != null) return false;
155            } else if (!mTagFormat.equals(other.mTagFormat)) return false;
156            if (mVersion == null) {
157                if (other.mVersion != null) return false;
158            } else if (!mVersion.equals(other.mVersion)) return false;
159            return true;
160        }
161    }
