005    package com.nuecho.rivr.voicexml.turn.output.fetch;
006    
007    import com.nuecho.rivr.core.util.*;
008    
009    /**
010     * @author Nu Echo Inc.
011     */
012    public abstract class BaseResourceFetchConfiguration {
013    
014        private final String mName;
015    
016        private TimeValue mMaxAge;
017        private TimeValue mMaxStale;
018        private FetchHint mFetchHint;
019    
020        protected BaseResourceFetchConfiguration(String name) {
021            mName = name;
022        }
023    
024        public final String getName() {
025            return mName;
026        }
027    
028        public final TimeValue getMaxAge() {
029            return mMaxAge;
030        }
031    
032        public final void setMaxAge(TimeValue maxAge) {
033            mMaxAge = maxAge;
034        }
035    
036        public final TimeValue getMaxStale() {
037            return mMaxStale;
038        }
039    
040        public final void setMaxStale(TimeValue maxStale) {
041            mMaxStale = maxStale;
042        }
043    
044        public final FetchHint getFetchHint() {
045            return mFetchHint;
046        }
047    
048        public final void setFetchHint(FetchHint fetchHint) {
049            mFetchHint = fetchHint;
050        }
051    
052        @Override
053        public int hashCode() {
054            final int prime = 31;
055            int result = 1;
056            result = prime * result + (mFetchHint == null ? 0 : mFetchHint.hashCode());
057            result = prime * result + (mMaxAge == null ? 0 : mMaxAge.hashCode());
058            result = prime * result + (mMaxStale == null ? 0 : mMaxStale.hashCode());
059            result = prime * result + (mName == null ? 0 : mName.hashCode());
060            return result;
061        }
062    
063        @Override
064        public boolean equals(Object obj) {
065            if (this == obj) return true;
066            if (obj == null) return false;
067            if (getClass() != obj.getClass()) return false;
068            BaseResourceFetchConfiguration other = (BaseResourceFetchConfiguration) obj;
069            if (mFetchHint == null) {
070                if (other.mFetchHint != null) return false;
071            } else if (!mFetchHint.equals(other.mFetchHint)) return false;
072            if (mMaxAge == null) {
073                if (other.mMaxAge != null) return false;
074            } else if (!mMaxAge.equals(other.mMaxAge)) return false;
075            if (mMaxStale == null) {
076                if (other.mMaxStale != null) return false;
077            } else if (!mMaxStale.equals(other.mMaxStale)) return false;
078            if (mName == null) {
079                if (other.mName != null) return false;
080            } else if (!mName.equals(other.mName)) return false;
081            return true;
082        }
083    }
