004    package com.nuecho.rivr.voicexml.turn.output.grammar;
005    
006    import javax.json.*;
007    
008    import com.nuecho.rivr.core.util.*;
009    import com.nuecho.rivr.voicexml.turn.output.fetch.*;
010    import com.nuecho.rivr.voicexml.util.json.*;
011    
012    /**
013     * A <code>GrammarReference</code> represents an external grammar reachable with
014     * an URI.
015     * 
016     * @author Nu Echo Inc.
017     * @see <a href="http://www.w3.org/TR/voicexml20/#dml3.1.1.2">http://www.w3.org/TR/voicexml20/#dml3.1.1.2</a>
018     */
019    public final class GrammarReference extends GrammarItem {
020        private static final String REFERENCE_ELEMENT_TYPE = "reference";
021        private static final String URI_PROPERTY = "uri";
022        private static final String RESOURCE_FETCH_CONFIGURATION_PROPERTY = "resourceFetchConfiguration";
023    
024        private final String mUri;
025        private ResourceFetchConfiguration mResourceFetchConfiguration;
026    
027        /**
028         * @param uri The URI of the external grammar. Not empty.
029         */
030        public GrammarReference(String uri) {
031            Assert.notEmpty(uri, "uri");
032            mUri = uri;
033        }
034    
035        /**
036         * @param resourceFetchConfiguration The resource fetch configuration. Null
037         *            reverts to VoiceXML default value.
038         */
039        public void setResourceFetchConfiguration(ResourceFetchConfiguration resourceFetchConfiguration) {
040            mResourceFetchConfiguration = resourceFetchConfiguration;
041        }
042    
043        @Override
044        public String getElementType() {
045            return REFERENCE_ELEMENT_TYPE;
046        }
047    
048        public String getUri() {
049            return mUri;
050        }
051    
052        public ResourceFetchConfiguration getResourceFetchConfiguration() {
053            return mResourceFetchConfiguration;
054        }
055    
056        @Override
057        protected void addJsonProperties(JsonObjectBuilder builder) {
058            JsonUtils.add(builder, URI_PROPERTY, mUri);
059            JsonUtils.add(builder, RESOURCE_FETCH_CONFIGURATION_PROPERTY, mResourceFetchConfiguration);
060        }
061    
062        @Override
063        public int hashCode() {
064            final int prime = 31;
065            int result = 1;
066            result = prime * result + (mResourceFetchConfiguration == null ? 0 : mResourceFetchConfiguration.hashCode());
067            result = prime * result + (mUri == null ? 0 : mUri.hashCode());
068            return result;
069        }
070    
071        @Override
072        public boolean equals(Object obj) {
073            if (this == obj) return true;
074            if (obj == null) return false;
075            if (getClass() != obj.getClass()) return false;
076            GrammarReference other = (GrammarReference) obj;
077            if (mResourceFetchConfiguration == null) {
078                if (other.mResourceFetchConfiguration != null) return false;
079            } else if (!mResourceFetchConfiguration.equals(other.mResourceFetchConfiguration)) return false;
080            if (mUri == null) {
081                if (other.mUri != null) return false;
082            } else if (!mUri.equals(other.mUri)) return false;
083            return true;
084        }
085    }
