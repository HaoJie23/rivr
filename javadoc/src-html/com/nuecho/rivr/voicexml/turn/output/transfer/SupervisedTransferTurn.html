008    import javax.json.*;

011    
012    import com.nuecho.rivr.core.util.*;

019    /**
020     * A <code>SupervisedTransferTurn</code> is a {@link TransferTurn} where the
021     * outcome is monitored and where a transfer failure can be handled gracefully.
022     * <p>
023     * Recognition can optionally be activated to cancel a supervised transfer.
024     * 
025     * @author Nu Echo Inc.
026     * @see BridgeTransferTurn
027     * @see ConsultationTransferTurn
028     */
029    public abstract class SupervisedTransferTurn extends TransferTurn {
030        private static final String SPEECH_RECOGNITION_CONFIGURATION_PROPERTY = "speechRecognitionConfiguration";
031        private static final String DTMF_RECOGNITION_CONFIGURATION_PROPERTY = "dtmfRecognitionConfiguration";
032        private static final String TRANSFER_AUDIO_PROPERTY = "transferAudio";
033        private static final String CONNECT_TIMEOUT_PROPERTY = "connectTimeout";
034    
035        private DtmfRecognitionConfiguration mDtmfRecognitionConfiguration;
036        private SpeechRecognitionConfiguration mSpeechRecognitionConfiguration;
037    
038        private Recording mTransferAudio;
039        private TimeValue mConnectTimeout;
040    
041        /**
042         * @param name The name of this turn. Not empty.
043         * @param destination The URI of the destination (telephone, IP telephony
044         *            address). Not empty.
045         */
046        public SupervisedTransferTurn(String name, String destination) {
047            super(name, destination);
048        }
049    
050        /**
051         * @param dtmfRecognitionConfiguration The active DTMF recognition
052         *            configuration during the transfer.
053         */
054        public final void setDtmfRecognitionConfiguration(DtmfRecognitionConfiguration dtmfRecognitionConfiguration) {
055            mDtmfRecognitionConfiguration = dtmfRecognitionConfiguration;
056        }
057    
058        /**
059         * @param speechRecognitionConfiguration The active speech recognition
060         *            configuration during the transfer.
061         */
062        public final void setSpeechRecognitionConfiguration(SpeechRecognitionConfiguration speechRecognitionConfiguration) {
063            mSpeechRecognitionConfiguration = speechRecognitionConfiguration;
064        }
065    
066        /**
067         * @param transferAudio The URI of audio source to play while the transfer
068         *            attempt is in progress.
069         */
070        public final void setTransferAudio(Recording transferAudio) {
071            mTransferAudio = transferAudio;
072        }
073    
074        /**
075         * @param connectTimeout The time to wait while trying to connect the call
076         *            before returning with {@link TransferStatus#NO_ANSWER}. Null
077         *            reverts to VoiceXML default value.
078         */
079        public final void setConnectTimeout(TimeValue connectTimeout) {
080            mConnectTimeout = connectTimeout;
081        }
082    
083        public final Recording getTransferAudio() {
084            return mTransferAudio;
085        }
086    
087        public final TimeValue getConnectTimeout() {
088            return mConnectTimeout;
089        }
090    
091        public final DtmfRecognitionConfiguration getDtmfRecognitionConfiguration() {
092            return mDtmfRecognitionConfiguration;
093        }
094    
095        public final SpeechRecognitionConfiguration getSpeechRecognitionConfiguration() {
096            return mSpeechRecognitionConfiguration;
097        }
098    
099        @Override
100        protected void customizeTransferElement(Element transferElement) throws VoiceXmlDocumentRenderingException {
101            VoiceXmlDomUtil.processDtmfConfiguration(mDtmfRecognitionConfiguration, transferElement);
102            VoiceXmlDomUtil.processSpeechConfiguration(mSpeechRecognitionConfiguration, transferElement);
103            setTimeAttribute(transferElement, CONNECT_TIMEOUT_ATTRIBUTE, mConnectTimeout);
104    
105            if (mTransferAudio != null) {
106                setAttribute(transferElement, TRANSFER_AUDIO_ATTRIBUTE, mTransferAudio.getPath());
107            }
108        }
109    
110        @Override
111        protected void addTurnProperties(JsonObjectBuilder builder) {
112            super.addTurnProperties(builder);
113            JsonUtils.addTimeProperty(builder, CONNECT_TIMEOUT_PROPERTY, mConnectTimeout);
114            JsonUtils.add(builder, TRANSFER_AUDIO_PROPERTY, mTransferAudio);
115            JsonUtils.add(builder, DTMF_RECOGNITION_CONFIGURATION_PROPERTY, mDtmfRecognitionConfiguration);
116            JsonUtils.add(builder, SPEECH_RECOGNITION_CONFIGURATION_PROPERTY, mSpeechRecognitionConfiguration);
117        }
118    }
