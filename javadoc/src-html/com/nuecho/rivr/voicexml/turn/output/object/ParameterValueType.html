004    package com.nuecho.rivr.voicexml.turn.output.object;
005    
006    import javax.json.*;
007    
008    import com.nuecho.rivr.voicexml.util.json.*;
009    
010    /**
011     * @author NuEcho Inc.
012     */
013    public enum ParameterValueType implements JsonSerializable {
014        DATA("data"), REF("ref");
015    
016        private final String mKey;
017    
018        private ParameterValueType(String key) {
019            mKey = key;
020        }
021    
022        public String getKey() {
023            return mKey;
024        }
025    
026        @Override
027        public String toString() {
028            return mKey;
029        }
030    
031        @Override
032        public JsonValue asJson() {
033            return JsonUtils.wrap(mKey);
034        }
035    }
