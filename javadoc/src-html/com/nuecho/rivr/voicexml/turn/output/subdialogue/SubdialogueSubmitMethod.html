004    package com.nuecho.rivr.voicexml.turn.output.subdialogue;
005    
006    import java.util.*;
007    
008    /**
009     * @author NuEcho Inc.
010     */
011    public final class SubdialogueSubmitMethod {
012        private static Map<String, SubdialogueSubmitMethod> sInstances = new HashMap<String, SubdialogueSubmitMethod>();
013    
014        public static final SubdialogueSubmitMethod GET = new SubdialogueSubmitMethod("get");
015        public static final SubdialogueSubmitMethod POST = new SubdialogueSubmitMethod("post");
016    
017        private final String mKey;
018    
019        private SubdialogueSubmitMethod(String key) {
020            mKey = key;
021            sInstances.put(key, this);
022        }
023    
024        public boolean isInternal() {
025            return false;
026        }
027    
028        public static SubdialogueSubmitMethod getMethodByName(String key) {
029            return sInstances.get(key);
030        }
031    
032        public static boolean exists(String key) {
033            return sInstances.containsKey(key);
034        }
035    
036        @Override
037        public String toString() {
038            return mKey;
039        }
040    
041        public String getKey() {
042            return mKey;
043        }
044    
045        public Object readResolve() {
046            return getMethodByName(mKey);
047        }
048    }
