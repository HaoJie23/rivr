004    package com.nuecho.rivr.voicexml.turn.output.transfer;
005    
006    /**
007     * A <code>ConsultationTransferTurn</code> is a is a
008     * {@link SupervisedTransferTurn} that is similar to a {@link BlindTransferTurn}
009     * except that the outcome of the transfer call setup is known and the caller is
010     * not dropped as a result of an unsuccessful transfer attempt.
011     * 
012     * @author Nu Echo Inc.
013     * @see <a href="http://www.w3.org/TR/voicexml21/#sec-xfer-consultation">http://www.w3.org/TR/voicexml21/#sec-xfer-consultation</a>
014     */
015    public class ConsultationTransferTurn extends SupervisedTransferTurn {
016        private static final String CONSULTATION_TRANSFER_TYPE = "consultation";
017    
018        /**
019         * @param name The name of this turn. Not empty.
020         * @param destination The URI of the destination (telephone, IP telephony
021         *            address). Not empty.
022         */
023        public ConsultationTransferTurn(String name, String destination) {
024            super(name, destination);
025        }
026    
027        @Override
028        protected final String getTransferType() {
029            return CONSULTATION_TRANSFER_TYPE;
030        }
031    }
