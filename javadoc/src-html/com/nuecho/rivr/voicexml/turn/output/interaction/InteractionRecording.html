005    package com.nuecho.rivr.voicexml.turn.output.interaction;
006    
007    import static java.util.Arrays.*;
008    
009    import java.util.*;
010    
011    import javax.json.*;
012    
013    import com.nuecho.rivr.core.util.*;
014    import com.nuecho.rivr.voicexml.turn.output.audio.*;
015    import com.nuecho.rivr.voicexml.util.json.*;
016    
017    /**
018     * An <code>InteractionRecording</code> is an optional final phase of an
019     * {@link InteractionTurn}.
020     * <p>
021     * It specifies a recording configuration, and optionally, a no input timeout
022     * and a sequence of {@link AudioItem} that is played if a recording is
023     * successful.
024     * 
025     * @author Nu Echo Inc.
026     */
027    public final class InteractionRecording implements JsonSerializable {
028        private static final String ACKNOWLEDGE_AUDIO_ITEMS_PROPERTY = "acknowledgeAudioItems";
029        private static final String RECORDING_CONFIGURATION_PROPERTY = "recordingConfiguration";
030        private static final String NO_INPUT_TIMEOUT_PROPERTY = "noInputTimeout";
031    
032        private final RecordingConfiguration mRecordingConfiguration;
033        private final TimeValue mNoInputTimeout;
034        private final List<AudioItem> mAcknowledgeAudioItems;
035    
036        public InteractionRecording(RecordingConfiguration recordingConfiguration,
037                                    TimeValue noInputTimeout,
038                                    AudioItem... acknowledgeAudioItems) {
039            this(recordingConfiguration, noInputTimeout, asList(acknowledgeAudioItems));
040        }
041    
042        public InteractionRecording(RecordingConfiguration recordingConfiguration,
043                                    TimeValue noInputTimeout,
044                                    List<? extends AudioItem> acknowledgeAudioItems) {
045            Assert.notNull(recordingConfiguration, "recordingConfiguration");
046            Assert.notNull(acknowledgeAudioItems, "acknowledgeAudioItems");
047            mRecordingConfiguration = recordingConfiguration;
048            mNoInputTimeout = noInputTimeout;
049            mAcknowledgeAudioItems = new ArrayList<AudioItem>(acknowledgeAudioItems);
050        }
051    
052        public RecordingConfiguration getRecordingConfiguration() {
053            return mRecordingConfiguration;
054        }
055    
056        public TimeValue getNoInputTimeout() {
057            return mNoInputTimeout;
058        }
059    
060        public List<? extends AudioItem> getAcknowledgeAudioItems() {
061            return mAcknowledgeAudioItems;
062        }
063    
064        @Override
065        public JsonValue asJson() {
066            JsonObjectBuilder builder = JsonUtils.createObjectBuilder();
067            JsonUtils.addTimeProperty(builder, NO_INPUT_TIMEOUT_PROPERTY, mNoInputTimeout);
068            JsonUtils.add(builder, RECORDING_CONFIGURATION_PROPERTY, mRecordingConfiguration);
069            JsonUtils.add(builder, ACKNOWLEDGE_AUDIO_ITEMS_PROPERTY, JsonUtils.toJson(mAcknowledgeAudioItems));
070            return builder.build();
071        }
072    
073        @Override
074        public String toString() {
075            return asJson().toString();
076        }
077    }
