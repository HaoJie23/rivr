005    package com.nuecho.rivr.voicexml.turn.output.interaction;
006    
007    import java.util.*;
008    
009    import javax.json.*;
010    
011    import com.nuecho.rivr.core.util.*;
012    import com.nuecho.rivr.voicexml.turn.output.grammar.*;
013    import com.nuecho.rivr.voicexml.util.json.*;
014    
015    /**
016     * A <code>SpeechRecognitionConfiguration</code> represents the speech part of
017     * an interaction phase recognition configuration. It is composed of a list of
018     * speech {@link GrammarItem} and optional properties (complete timeout,
019     * incomplete timeout, max speech timeout, max nbest, confidence level, speed
020     * versus accuracy and sensitivity).
021     * 
022     * @author Nu Echo Inc.
023     */
024    public final class SpeechRecognitionConfiguration extends RecognitionConfiguration {
025    
026        private static final String SPEED_VERSUS_ACCURACY_PROPERTY = "speedVersusAccuracy";
027        private static final String SENSITIVITY_PROPERTY = "sensitivity";
028        private static final String MAX_N_BEST_PROPERTY = "maxNBest";
029        private static final String CONFIDENCE_LEVEL_PROPERTY = "confidenceLevel";
030        private static final String MAX_SPEECH_TIMEOUT_PROPERTY = "maxSpeechTimeout";
031        private static final String INCOMPLETE_TIMEOUT_PROPERTY = "incompleteTimeout";
032        private static final String COMPLETE_TIMEOUT_PROPERTY = "completeTimeout";
033        private TimeValue mCompleteTimeout;
034        private TimeValue mIncompleteTimeout;
035        private TimeValue mMaxSpeechTimeout;
036        private Integer mMaxNBest;
037    
038        private Double mConfidenceLevel;
039        private Double mSpeedVersusAccuracy;
040        private Double mSensitivity;
041    
042        /**
043         * @param grammarItems The list of speech {@link GrammarItem}. Not null.
044         */
045        public SpeechRecognitionConfiguration(GrammarItem... grammarItems) {
046            super(grammarItems);
047        }
048    
049        /**
050         * @param grammarItems The list of speech {@link GrammarItem}. Not null.
051         */
052        public SpeechRecognitionConfiguration(List<GrammarItem> grammarItems) {
053            super(grammarItems);
054        }
055    
056        /**
057         * @param completeTimeout The required length of silence following user
058         *            speech before the speech recognizer finalizes a complete
059         *            result. Null reverts to VoiceXML default value.
060         * @see <a href="http://www.w3.org/TR/voicexml20/#dml6.3.2">http://www.w3.org/TR/voicexml20/#dml6.3.2</a>
061         */
062        public void setCompleteTimeout(TimeValue completeTimeout) {
063            mCompleteTimeout = completeTimeout;
064        }
065    
066        /**
067         * @param incompleteTimeout The required length of silence following user
068         *            speech after which a recognizer finalizes an incomplete
069         *            result. Null reverts to VoiceXML default value.
070         * @see <a href="http://www.w3.org/TR/voicexml20/#dml6.3.2">http://www.w3.org/TR/voicexml20/#dml6.3.2</a>
071         */
072        public void setIncompleteTimeout(TimeValue incompleteTimeout) {
073            mIncompleteTimeout = incompleteTimeout;
074        }
075    
076        /**
077         * @param maxSpeechTimeout The maximum duration of user speech. Null reverts
078         *            to VoiceXML default value.
079         * @see <a href="http://www.w3.org/TR/voicexml20/#dml6.3.2">http://www.w3.org/TR/voicexml20/#dml6.3.2</a>
080         */
081        public void setMaxSpeechTimeout(TimeValue maxSpeechTimeout) {
082            mMaxSpeechTimeout = maxSpeechTimeout;
083        }
084    
085        /**
086         * @param maxNBest The maximum size of the recognition result. Must be a
087         *            positive integer. Null reverts to VoiceXML default value.
088         * @see <a href="http://www.w3.org/TR/voicexml20/#dml6.3.2">http://www.w3.org/TR/voicexml20/#dml6.3.2</a>
089         */
090        public void setMaxNBest(Integer maxNBest) {
091            if (maxNBest != null) {
092                Assert.positive(maxNBest, "maxNBest");
093            }
094    
095            mMaxNBest = maxNBest;
096        }
097    
098        /**
099         * @param confidenceLevel The speech recognition confidence level. Results
100         *            with confidence below this value will be rejected (nomatch).
101         *            Value must be between 0.0 and 1.0. Null reverts to VoiceXML
102         *            default value.
103         * @see <a href="http://www.w3.org/TR/voicexml20/#dml6.3.2">http://www.w3.org/TR/voicexml20/#dml6.3.2</a>
104         */
105        public void setConfidenceLevel(Double confidenceLevel) {
106            if (confidenceLevel != null) {
107                Assert.between(0, confidenceLevel, 1.0);
108            }
109    
110            mConfidenceLevel = confidenceLevel;
111        }
112    
113        /**
114         * @param speedVersusAccuracy A hint specifying the desired balance between
115         *            speed versus accuracy. A value of 0.0 means fastest
116         *            recognition. A value of 1.0 means best accuracy. Null reverts
117         *            to VoiceXML default value.
118         * @see <a href="http://www.w3.org/TR/voicexml20/#dml6.3.2">http://www.w3.org/TR/voicexml20/#dml6.3.2</a>
119         */
120        public void setSpeedVersusAccuracy(Double speedVersusAccuracy) {
121            if (speedVersusAccuracy != null) {
122                Assert.between(0, speedVersusAccuracy, 1.0);
123            }
124    
125            mSpeedVersusAccuracy = speedVersusAccuracy;
126        }
127    
128        /**
129         * @param sensitivity Set the sensitivity level. A value of 1.0 means that
130         *            it is highly sensitive to quiet input. A value of 0.0 means it
131         *            is least sensitive to noise. Null reverts to VoiceXML default
132         *            value.
133         */
134        public void setSensitivity(Double sensitivity) {
135            if (sensitivity != null) {
136                Assert.between(0, sensitivity, 1.0);
137            }
138    
139            mSensitivity = sensitivity;
140        }
141    
142        public TimeValue getCompleteTimeout() {
143            return mCompleteTimeout;
144        }
145    
146        public TimeValue getIncompleteTimeout() {
147            return mIncompleteTimeout;
148        }
149    
150        public TimeValue getMaxSpeechTimeout() {
151            return mMaxSpeechTimeout;
152        }
153    
154        public Integer getMaxNBest() {
155            return mMaxNBest;
156        }
157    
158        public Double getConfidenceLevel() {
159            return mConfidenceLevel;
160        }
161    
162        public Double getSpeedVersusAccuracy() {
163            return mSpeedVersusAccuracy;
164        }
165    
166        public Double getSensitivity() {
167            return mSensitivity;
168        }
169    
170        @Override
171        protected void addJsonProperties(JsonObjectBuilder builder) {
172            JsonUtils.addTimeProperty(builder, COMPLETE_TIMEOUT_PROPERTY, mCompleteTimeout);
173            JsonUtils.addTimeProperty(builder, INCOMPLETE_TIMEOUT_PROPERTY, mIncompleteTimeout);
174            JsonUtils.addTimeProperty(builder, MAX_SPEECH_TIMEOUT_PROPERTY, mMaxSpeechTimeout);
175            JsonUtils.addNumberProperty(builder, CONFIDENCE_LEVEL_PROPERTY, mConfidenceLevel);
176            JsonUtils.addNumberProperty(builder, MAX_N_BEST_PROPERTY, mMaxNBest);
177            JsonUtils.addNumberProperty(builder, SENSITIVITY_PROPERTY, mSensitivity);
178            JsonUtils.addNumberProperty(builder, SPEED_VERSUS_ACCURACY_PROPERTY, mSpeedVersusAccuracy);
179        }
180    
181        @Override
182        public int hashCode() {
183            final int prime = 31;
184            int result = 1;
185            result = prime * result + (mCompleteTimeout == null ? 0 : mCompleteTimeout.hashCode());
186            result = prime * result + (mMaxNBest == null ? 0 : mMaxNBest.hashCode());
187            result = prime * result + (mIncompleteTimeout == null ? 0 : mIncompleteTimeout.hashCode());
188            result = prime * result + (mMaxSpeechTimeout == null ? 0 : mMaxSpeechTimeout.hashCode());
189            result = prime * result + (mConfidenceLevel == null ? 0 : mConfidenceLevel.hashCode());
190            result = prime * result + (mSensitivity == null ? 0 : mSensitivity.hashCode());
191            result = prime * result + (mSpeedVersusAccuracy == null ? 0 : mSpeedVersusAccuracy.hashCode());
192            return result;
193        }
194    
195        @Override
196        public boolean equals(Object obj) {
197            if (this == obj) return true;
198            if (obj == null) return false;
199            if (getClass() != obj.getClass()) return false;
200            SpeechRecognitionConfiguration other = (SpeechRecognitionConfiguration) obj;
201            if (mCompleteTimeout == null) {
202                if (other.mCompleteTimeout != null) return false;
203            } else if (!mCompleteTimeout.equals(other.mCompleteTimeout)) return false;
204            if (mMaxNBest == null) {
205                if (other.mMaxNBest != null) return false;
206            } else if (!mMaxNBest.equals(other.mMaxNBest)) return false;
207            if (mIncompleteTimeout == null) {
208                if (other.mIncompleteTimeout != null) return false;
209            } else if (!mIncompleteTimeout.equals(other.mIncompleteTimeout)) return false;
210            if (mMaxSpeechTimeout == null) {
211                if (other.mMaxSpeechTimeout != null) return false;
212            } else if (!mMaxSpeechTimeout.equals(other.mMaxSpeechTimeout)) return false;
213            if (mConfidenceLevel == null) {
214                if (other.mConfidenceLevel != null) return false;
215            } else if (!mConfidenceLevel.equals(other.mConfidenceLevel)) return false;
216            if (mSensitivity == null) {
217                if (other.mSensitivity != null) return false;
218            } else if (!mSensitivity.equals(other.mSensitivity)) return false;
219            if (mSpeedVersusAccuracy == null) {
220                if (other.mSpeedVersusAccuracy != null) return false;
221            } else if (!mSpeedVersusAccuracy.equals(other.mSpeedVersusAccuracy)) return false;
222            return true;
223        }
224    
225        public SpeechRecognitionConfiguration copy() {
226            SpeechRecognitionConfiguration copy = new SpeechRecognitionConfiguration(getGrammarItems());
227            copy.mCompleteTimeout = mCompleteTimeout;
228            copy.mMaxNBest = mMaxNBest;
229            copy.mIncompleteTimeout = mIncompleteTimeout;
230            copy.mMaxSpeechTimeout = mMaxSpeechTimeout;
231            copy.mConfidenceLevel = mConfidenceLevel;
232            copy.mSensitivity = mSensitivity;
233            copy.mSpeedVersusAccuracy = mSpeedVersusAccuracy;
234            copyPropertiesTo(copy);
235            return copy;
236        }
237    }
