023    /**
024     * A <code>SubdialogueInvocationTurn</code> is a {@link VoiceXmlOutputTurn} that
025     * invokes another external dialogue.
026     * <p>
027     * Parameters may be passed to the subdialogue and return values may be
028     * retrieved if the invoked subdialogue ends with a <code>&lt;return&gt;</code>
029     * instruction.
030     * 
031     * @author Nu Echo Inc.
032     * @see SubdialogueParameter
033     * @see SubdialogueSubmitMethod
034     * @see <a
035     *      href="http://www.w3.org/TR/voicexml20/#dml2.3.4">http://www.w3.org/TR/voicexml20/#dml2.3.4</a>
036     */
037    public class SubdialogueInvocationTurn extends VoiceXmlOutputTurn {
038        public static final String SUBDIALOGUE_RESULT_VARIABLE_NAME = "subdialogue";
039    
040        private static final String SUBDIALOGUE_INVOCATION_TURN_TYPE = "subdialogue";
041    
042        private static final String POST_DIALOGUE_SCRIPT_PROPERTY = "postDialogueScript";
043        private static final String SUBDIALOGUE_PARAMETERS_PROPERTY = "subdialogueParameters";
044        private static final String SUBMIT_URI_PROPERTY = "uri";
045        private static final String SUBMIT_PARAMETERS_PROPERTY = "submitParameters";
046        private static final String SUBMIT_METHOD_PROPERTY = "submitMethod";
047        private static final String SUBDIALOGUE_FETCH_CONFIGURATION_PROPERTY = "subdialogueFetchConfiguration";
048    
049        private final String mUri;
050        private List<SubdialogueParameter> mSubdialogueParameters = Collections.emptyList();
051        private VariableDeclarationList mSubmitParameters = new VariableDeclarationList();
052        private SubdialogueSubmitMethod mMethod = SubdialogueSubmitMethod.GET;
053        private SubdialogueFetchConfiguration mSubdialogueFetchConfiguration;
054        private String mPostDialogueScript;
055    
056        /**
057         * @param name The name of this turn. Not empty.
058         * @param uri The URI of the subdialogue. Not empty.
059         */
060        public SubdialogueInvocationTurn(String name, String uri) {
061            super(name);
062            Assert.notEmpty(uri, "uri");
063            mUri = uri;
064        }
065    
066        /**
067         * @param subdialogueParameters A list of {@link SubdialogueParameter} that
068         *            will be passed to the subdialogue. Not null.
069         */
070        public final void setSubdialogueParameters(List<SubdialogueParameter> subdialogueParameters) {
071            Assert.notNull(subdialogueParameters, "subdialogueParameters");
072            mSubdialogueParameters = new ArrayList<SubdialogueParameter>(subdialogueParameters);
073        }
074    
075        /**
076         * @param subdialogueParameters A list of {@link SubdialogueParameter} that
077         *            will be passed to the subdialogue. Not null.
078         */
079        public final void setSubdialogueParameters(SubdialogueParameter... subdialogueParameters) {
080            setSubdialogueParameters(asList(subdialogueParameters));
081        }
082    
083        /**
084         * @param submitParameters A list of variable to submit when invoking the
085         *            URI. Not null.
086         */
087        public final void setSubmitParameters(VariableDeclarationList submitParameters) {
088            Assert.notNull(submitParameters, "submitParameters");
089            mSubmitParameters = submitParameters;
090        }
091    
092        /**
093         * @param method The HTTP method used to invoke the subdialogue (GET or
094         *            POST). Null reverts to VoiceXML default value.
095         * @see SubdialogueSubmitMethod
096         */
097        public final void setMethod(SubdialogueSubmitMethod method) {
098            mMethod = method;
099        }
100    
101        /**
102         * @param subdialogueFetchConfiguration The
103         *            {@link SubdialogueFetchConfiguration}. Null reverts to
104         *            VoiceXML default value.
105         */
106        public final void setSubdialogueFetchConfiguration(SubdialogueFetchConfiguration subdialogueFetchConfiguration) {
107            mSubdialogueFetchConfiguration = subdialogueFetchConfiguration;
108        }
109    
110        /**
111         * @param postDialogueScript The ECMAScript script to execute after
112         *            subdialogue invocation.
113         */
114        public final void setPostDialogueScript(String postDialogueScript) {
115            mPostDialogueScript = postDialogueScript;
116        }
117    
118        public final String getUri() {
119            return mUri;
120        }
121    
122        public final List<? extends SubdialogueParameter> getVoiceXmlParameters() {
123            return unmodifiableList(mSubdialogueParameters);
124        }
125    
126        public final VariableDeclarationList getSubmitParameters() {
127            return mSubmitParameters;
128        }
129    
130        public final SubdialogueSubmitMethod getMethod() {
131            return mMethod;
132        }
133    
134        public final String getPostDialogueScript() {
135            return mPostDialogueScript;
136        }
137    
138        public final SubdialogueFetchConfiguration getSubdialogueFetchConfiguration() {
139            return mSubdialogueFetchConfiguration;
140        }
141    
142        @Override
143        protected final String getOuputTurnType() {
144            return SUBDIALOGUE_INVOCATION_TURN_TYPE;
145        }
146    
147        @Override
148        protected void addTurnProperties(JsonObjectBuilder builder) {
149            JsonUtils.add(builder, SUBMIT_URI_PROPERTY, mUri);
150            JsonUtils.add(builder, SUBMIT_METHOD_PROPERTY, mMethod.getKey());
151            JsonUtils.add(builder, SUBMIT_PARAMETERS_PROPERTY, mSubmitParameters);
152            JsonUtils.add(builder, SUBDIALOGUE_PARAMETERS_PROPERTY, JsonUtils.toJson(mSubdialogueParameters));
153            JsonUtils.add(builder, SUBDIALOGUE_FETCH_CONFIGURATION_PROPERTY, mSubdialogueFetchConfiguration);
154            JsonUtils.add(builder, POST_DIALOGUE_SCRIPT_PROPERTY, getPostDialogueScript());
155        }
156    
157        @Override
158        protected void fillVoiceXmlDocument(Document document, Element formElement, VoiceXmlDialogueContext dialogueContext)
159                throws VoiceXmlDocumentRenderingException {
160    
161            List<String> submitNameList = new ArrayList<String>();
162            VariableDeclarationList submitVariableList = mSubmitParameters;
163            if (submitVariableList != null) {
164                addVariableDeclarations(formElement, submitVariableList);
165    
166                for (VariableDeclaration declaration : mSubmitParameters) {
167                    submitNameList.add(declaration.getName());
168                }
169            }
170    
171            Element subdialogueElement = DomUtils.appendNewElement(formElement, SUBDIALOG_ELEMENT);
172            subdialogueElement.setAttribute(NAME_ATTRIBUTE, SUBDIALOGUE_FORM_ITEM_NAME);
173            subdialogueElement.setAttribute(SRC_ATTRIBUTE, mUri);
174    
175            if (!submitNameList.isEmpty()) {
176                subdialogueElement.setAttribute(NAME_LIST_ATTRIBUTE, StringUtils.join(submitNameList, " "));
177            }
178    
179            for (SubdialogueParameter subdialogueParameter : mSubdialogueParameters) {
180                Element paramElement = DomUtils.appendNewElement(subdialogueElement, PARAM_ELEMENT);
181                paramElement.setAttribute(NAME_ATTRIBUTE, subdialogueParameter.getName());
182                setAttribute(paramElement, VALUE_ATTRIBUTE, subdialogueParameter.getValue());
183                setAttribute(paramElement, EXPR_ATTRIBUTE, subdialogueParameter.getExpression());
184            }
185    
186            SubdialogueSubmitMethod submitMethod = mMethod;
187            if (submitMethod != null) {
188                subdialogueElement.setAttribute(METHOD_ATTRIBUTE, submitMethod.getKey());
189            }
190    
191            SubdialogueFetchConfiguration fetchConfiguration = mSubdialogueFetchConfiguration;
192            if (fetchConfiguration != null) {
193                applyFetchAudio(subdialogueElement, fetchConfiguration.getFetchAudio());
194                applyRessourceFetchConfiguration(subdialogueElement, fetchConfiguration.getResourceFetchConfiguration());
195            }
196    
197            Element filledElement = DomUtils.appendNewElement(subdialogueElement, FILLED_ELEMENT);
198    
199            createVarElement(filledElement, SUBDIALOGUE_RESULT_VARIABLE_NAME, "dialog." + SUBDIALOGUE_FORM_ITEM_NAME);
200    
201            if (mPostDialogueScript != null) {
202                createScript(filledElement, mPostDialogueScript);
203            }
204    
205            createScript(filledElement, RIVR_SCOPE_OBJECT + ".addValueResult(" + SUBDIALOGUE_RESULT_VARIABLE_NAME + ");");
206            createGotoSubmit(filledElement);
207        }
