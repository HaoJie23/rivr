005    package com.nuecho.rivr.voicexml.turn.output.interaction;
006    
007    import javax.json.*;
008    
009    import com.nuecho.rivr.core.util.*;
010    import com.nuecho.rivr.voicexml.turn.input.*;
011    import com.nuecho.rivr.voicexml.util.json.*;
012    
013    /**
014     * A <code>RecordingConfiguration</code> represents the description of a
015     * recording final phase of an interaction.
016     * <p>
017     * It may have an optional {@link DtmfRecognitionConfiguration} to recognize
018     * DTMF input while recording. A DTMF input matching one of the
019     * {@link DtmfRecognitionConfiguration} grammar will terminate the recording and
020     * place the recording in a variable.
021     * 
022     * @author Nu Echo Inc.
023     * @see DtmfRecognitionConfiguration
024     * @see <a href="http://www.w3.org/TR/voicexml20/#dml2.3.6">http://www.w3.org/TR/voicexml20/#dml2.3.6</a>
025     */
026    public final class RecordingConfiguration implements JsonSerializable {
027        private static final String DTMF_TERM_PROPERTY = "dtmfTerm";
028        private static final String BEEP_PROPERTY = "beep";
029        private static final String POST_AUDIO_TO_SERVER_PROPERTY = "postAudioToServer";
030        private static final String MAXIMUM_TIME_PROPERTY = "maximumTime";
031        private static final String FINAL_SILENCE_PROPERTY = "finalSilence";
032        private static final String DTMF_TERM_RECOGNITION_CONFIGURATION_PROPERTY = "dtmfTermRecognitionConfiguration";
033        private static final String CLIENT_SIDE_ASSIGNATION_DESTINATION_PROPERTY = "clientSideAssignationDestination";
034        private static final String TYPE_PROPERTY = "type";
035    
036        private DtmfRecognitionConfiguration mDtmfTermRecognitionConfiguration;
037        private Boolean mBeep;
038        private TimeValue mMaximumTime;
039        private TimeValue mFinalSilence;
040        private String mType;
041        private String mClientSideAssignationDestination;
042        private Boolean mDtmfTerm;
043        private boolean mPostAudioToServer = true;
044    
045        /**
046         * @param dtmfRecognitionConfiguration The
047         *            {@link DtmfRecognitionConfiguration} used to interrupt the
048         *            recording. Null reverts to VoiceXML default value.
049         */
050        public void setDtmfTermRecognitionConfiguration(DtmfRecognitionConfiguration dtmfRecognitionConfiguration) {
051            mDtmfTermRecognitionConfiguration = dtmfRecognitionConfiguration;
052        }
053    
054        /**
055         * @param beep If true, a tone is emitted just prior to recording. Null
056         *            reverts to VoiceXML default value.
057         */
058        public void setBeep(Boolean beep) {
059            mBeep = Boolean.valueOf(beep);
060        }
061    
062        /**
063         * @param maximumTime The maximum duration to record. Null reverts to
064         *            VoiceXML default value.
065         */
066        public void setMaximumTime(TimeValue maximumTime) {
067            mMaximumTime = maximumTime;
068        }
069    
070        /**
071         * @param finalSilence The interval of silence that indicates end of speech.
072         *            Null reverts to VoiceXML default value.
073         */
074        public void setFinalSilence(TimeValue finalSilence) {
075            mFinalSilence = finalSilence;
076        }
077    
078        /**
079         * @param type The media format of the resulting recording. Null reverts to
080         *            VoiceXML default value.
081         * @see <a href="http://www.w3.org/TR/voicexml20/#dmlAAudioFormats">http://www.w3.org/TR/voicexml20/#dmlAAudioFormats</a>
082         */
083        public void setType(String type) {
084            mType = type;
085        }
086    
087        /**
088         * @param clientSideAssignationDestination The variable where the recording
089         *            will be stored.
090         */
091        public void setClientSideAssignationDestination(String clientSideAssignationDestination) {
092            mClientSideAssignationDestination = clientSideAssignationDestination;
093        }
094    
095        /**
096         * @param dtmfTerm If true, any DTMF keypress not matched by an active
097         *            grammar will be treated as a match of an active local DTMF
098         *            grammar. Null reverts to VoiceXML default value.
099         */
100        public void setDtmfTerm(Boolean dtmfTerm) {
101            mDtmfTerm = Boolean.valueOf(dtmfTerm);
102        }
103    
104        /**
105         * @param postAudioToServer If true, recording will be posted to server and
106         *            made available in the {@link RecordingData} property of the
107         *            {@link RecordingInfo}.
108         */
109        public void setPostAudioToServer(boolean postAudioToServer) {
110            mPostAudioToServer = postAudioToServer;
111        }
112    
113        public DtmfRecognitionConfiguration getDtmfTermRecognitionConfiguration() {
114            return mDtmfTermRecognitionConfiguration;
115        }
116    
117        public Boolean getBeep() {
118            return mBeep;
119        }
120    
121        public TimeValue getMaximumTime() {
122            return mMaximumTime;
123        }
124    
125        public TimeValue getFinalSilence() {
126            return mFinalSilence;
127        }
128    
129        public Boolean getDtmfTerm() {
130            return mDtmfTerm;
131        }
132    
133        public String getType() {
134            return mType;
135        }
136    
137        public String getClientSideAssignationDestination() {
138            return mClientSideAssignationDestination;
139        }
140    
141        public boolean isPostAudioToServer() {
142            return mPostAudioToServer;
143        }
144    
145        @Override
146        public JsonValue asJson() {
147            JsonObjectBuilder builder = JsonUtils.createObjectBuilder();
148            JsonUtils.addBooleanProperty(builder, BEEP_PROPERTY, mBeep);
149            JsonUtils.addBooleanProperty(builder, DTMF_TERM_PROPERTY, mDtmfTerm);
150            JsonUtils.addBooleanProperty(builder, POST_AUDIO_TO_SERVER_PROPERTY, mPostAudioToServer);
151            JsonUtils.addTimeProperty(builder, MAXIMUM_TIME_PROPERTY, mMaximumTime);
152            JsonUtils.addTimeProperty(builder, FINAL_SILENCE_PROPERTY, mFinalSilence);
153            JsonUtils.add(builder, DTMF_TERM_RECOGNITION_CONFIGURATION_PROPERTY, mDtmfTermRecognitionConfiguration);
154            JsonUtils.add(builder, TYPE_PROPERTY, mType);
155            JsonUtils.add(builder, CLIENT_SIDE_ASSIGNATION_DESTINATION_PROPERTY, mClientSideAssignationDestination);
156            return builder.build();
157        }
158    
159        @Override
160        public String toString() {
161            return asJson().toString();
162        }
163    }
